  begin
    readln(n);

      for j:=1 to n do
        begin
          read(ga[i,j]);
          if ga[i,j]=0 then ga[i,j]:=maxlongint;
        end;

      begin
        l[i].front:=1;
        l[i].rear:=i+1;

      end;
  end;
 
procedure work;
  var i,j,k,min,minum,ans:longint;
      temp:node;
  begin
    ans:=0;
    for i:=1 to n-1 do
      begin
        min:=maxlongint;
        minum:=0;
        for j:=i to n-1 do
          if l[j].weight<min then begin min:=l[j].weight; minum:=j; end;
        if minum<>0 then
          begin
            temp:=l[i];
            l[i]:=l[minum];
            l[minum]:=temp;
            ans:=ans+min;
            k:=l[i].rear;

              if ga[k,l[j].rear]<l[j].weight then begin l[j].weight:=ga[k,l[j].rear]; l[j].front:=k; end;
          end;
      end;
    writeln(ans);
  end;

Var
 n,i,j,k,p,ans,edge:longint;
 e:array[1..10000] of longint;
 x,y:array[1..10000] of byte;
 dad:array[1..100] of byte;
 
Procedure opf;
begin
 assign(input,'agrinet.in');
 assign(output,'agrinet.out');
 reset(input); rewrite(output);
end;
 
Procedure clf;
begin
 close(input); close(output);
end;
 
Function find(x:longint):longint;
begin
 if dad[x]=x then exit(x);
 dad[x]:=find(dad[x]);
 exit(dad[x]);
end;
 
Procedure union(x,y:longint);
var fx,fy:longint;
begin
 fx:=find(x);
 fy:=find(y);
 if fx<>fy then dad[fx]:=fy;
end;
 
Procedure qsort(p,q:longint);
var l,r,mid,t:longint;
begin
 l:=p; r:=q; mid:=e[random(r-l)+l];
 repeat
  while not(e[l]>=mid) do inc(l);
  while not(e[r]<=mid) do dec(r);
  if not(l>r) then begin
    t:=e[l]; e[l]:=e[r]; e[r]:=t;
    t:=x[l]; x[l]:=x[r]; x[r]:=t;
    t:=y[l]; y[l]:=y[r]; y[r]:=t;
    inc(l); dec(r);
   end;
 until l>r;
 if l<q then qsort(l,q);
 if p<r then qsort(p,r);
end;
 
Begin
 opf;
 
 readln(n);
 for i:=1 to n do dad[i]:=i;
 
 for j:=1 to n do
  for i:=1 to n do begin
    read(k);
    if i<j then begin
      inc(edge);
      e[edge]:=k;
      x[edge]:=i;
      y[edge]:=j;
     end;
   end;
 
 randomize;
 qsort(1,edge);
 
 i:=0; k:=0;
 repeat
  inc(i);
  if find(x[i])<>find(y[i]) then begin
    inc(ans,e[i]);
    union(x[i],y[i]);
    inc(k);
   end;
 until k=n-1;
 
 writeln(ans);
 
 clf;
End.
