program buylow;
type
  aaa=array[1..200] of longint;
  data=record
    val,long:longint;
    num:aaa;
  end;
var
  i,j,k,m,n:longint;
  ago,a:array[1..5000] of longint;
  f:array[0..5000] of data;
  ans:data;
begin
  assign(input,'buylow.in'); assign(output,'buylow.out');
  reset(input); rewrite(output);
  fillchar(ago,sizeof(ago),0); fillchar(a,sizeof(a),0); fillchar(f,sizeof(f),0);
  read(n);
  for i:=1 to n do
  begin
    read(a[i]);
    ago[i]:=-1;
    for j:=i-1 downto 1 do
     if a[i]=a[j] then
      begin
       ago[i]:=j; break;
      end;
  end;
  a[0]:=maxlongint div 2; f[0].num[1]:=1; f[0].long:=1;
  for i:=1 to n do
  begin
    for j:=ago[i]+1 to i-1 do
     if a[i]<a[j] then
      begin
        if f[j].val+1>f[i].val then
        begin
          f[i]:=f[j]; f[i].val:=f[j].val+1; continue;
        end;
        if f[j].val+1=f[i].val then
         begin
           m:=f[j].long;
           if f[i].long>m then m:=f[i].long;
           inc(f[i].num[1],f[j].num[1]);
           for k:=2 to m do
           begin
             inc(f[i].num[k],f[j].num[k]+f[i].num[k-1] div 100000000);
             f[i].num[k-1]:=f[i].num[k-1]  mod 100000000;
           end;
           while f[i].num[m]>99999999 do
           begin
             inc(m);
             inc(f[i].num[m],f[i].num[m-1] div 100000000);
             f[i].num[m-1]:=f[i].num[m-1]  mod 100000000;
           end;
           f[i].long:=m;
         end;
      end;
  end;
  fillchar(ans,sizeof(ans),0);
  ans.val:=0; ans.long:=1;
  for i:=1 to n do
   if f[i].val>ans.val then ans.val:=f[i].val;
  for i:=n downto 1 do
   if f[i].val=ans.val then break;
  while i>0 do
  begin
    j:=ans.long;
    if ans.long<f[i].long then j:=f[i].long;
    ans.long:=j;
    ans.num[1]:=ans.num[1]+f[i].num[1];
    for k:=2 to j do
    begin
     ans.num[k]:=ans.num[k]+f[i].num[k]+ans.num[k-1] div 100000000;
     ans.num[k-1]:=ans.num[k-1] mod 100000000;
    end;
    while ans.num[ans.long]>99999999 do
    begin
      inc(ans.long);
      ans.num[ans.long]:=ans.num[ans.long]+ans.num[ans.long-1] div 100000000;
      ans.num[ans.long-1]:=ans.num[ans.long-1] mod 100000000;
    end;
    while i>0 do
    begin
      dec(i);
      if f[i].val=ans.val then break;
    end;
  end;
  write(ans.val,' ');
  write(ans.num[ans.long]);
  for i:=ans.long-1 downto 1 do
  begin
    if ans.num[i]<10000000 then write(0);
    if ans.num[i]<1000000 then write(0);
    if ans.num[i]<100000 then write(0);
    if ans.num[i]<10000 then write(0);
    if ans.num[i]<1000 then write(0);
    if ans.num[i]<100 then write(0);
    if ans.num[i]<10 then write(0);
    write(ans.num[i]);
  end;
  writeln;
  close(input); close(output);
end.

function find(l,r,d:longint):longint;
var mid:longint;
begin
if l=r then exit(l);
mid:=(l+r) div 2;
if d<=h[mid] then exit(find(l,mid,d));
if d>h[mid]  then exit(find(mid+1,r,d));

Program buylow;
Const
  inp='buylow.in';
  oup='buylow.out';
  maxn=5000;
Type
  num=array[0..100] of integer;
Var
  dp:array[1..maxn+1] of integer;
  s:array[1..maxn+1] of longint;
  source:array[1..maxn+1] of num;
  start:array[1..maxn+1] of longint;
  n,i,j,last:integer;
Procedure add(a,b:integer);
  var
    i,w:integer;
  begin
    w:=source[a][0];
    if w<source[b][0] then w:=source[b][0];
    for i:=1 to w do
      inc(source[a][i],source[b][i]);
    for i:=1 to w-1 do
      if source[a][i]>=10 then
        begin
          inc(source[a][i+1],source[a][i] div 10);
          source[a][i]:=source[a][i] mod 10;
        end;
    if source[a][w]>=10 then
      begin
        source[a][w+1]:=source[a][w] div 10;
        source[a][w]:=source[a][w] mod 10;
        inc(w);
      end;
    source[a][0]:=w;
  end;
 
Procedure dpa;
  var
    i,j:integer;
  begin
    for i:=2 to n do
      for j:=start[i] to i-1 do
        if s[i]<s[j] then
          begin
            if dp[i]<dp[j]+1 then
              begin
                dp[i]:=dp[j]+1;
                source[i]:=source[j];
              end
            else if dp[i]=dp[j]+1 then
              add(i,j);
          end;
    for i:=1 to n do
      if dp[i]>dp[n+1] then
        begin
          dp[n+1]:=dp[i];
          source[n+1]:=source[i];
        end
      else if dp[i]=dp[n+1] then add(n+1,i);
  end;
 
Begin
  assign(input,inp); reset(input);

  for i:=1 to n do
    begin
      read(s[i]);
      if eoln then readln;
      last:=0;

        if s[j]=s[i] then
          last:=j;
      if last=0 then dp[i]:=1 else dp[i]:=-maxint;
      source[i][0]:=1;
      source[i][1]:=1;
      start[i]:=last+1;
    end;
  close(input);
  dpa;
  assign(output,oup); rewrite(output);
  write(dp[n+1],' ');
  for i:=source[n+1][0] downto 2 do write(source[n+1][i]);
