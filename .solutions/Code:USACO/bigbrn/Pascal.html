program bigbrn;
var i,j,n,t,a,b,ans:longint;
    pd:array[0..1000,0..1000]of boolean;
    f:array[0..1000,0..1000]of longint;
function min(a,b,c:longint):longint;
begin
    if a>b then a:=b; if a>c then a:=c; exit(a);
end;
begin
    assign(input,'bigbrn.in');reset(input);
    assign(output,'bigbrn.out');rewrite(output);
    readln(n,t);  fillchar(pd,sizeof(pd),true);
    for i:=1 to t do
    begin
        readln(a,b);
        pd[a,b]:=false;
    end;
    for i:=1 to n do
      for j:=1 to n do
      if pd[i,j] then
      begin
          f[i,j]:=min(f[i-1,j],f[i,j-1],f[i-1,j-1])+1;
          if f[i,j]>ans then ans:=f[i,j];
      end;
    writeln(ans);
    close(input);close(output);
end.

                                                  up[i,j]记录(i,j)上边，包括(i,j)的最大连续长度；}

begin
  if a<b then min:=a else min:=b;
