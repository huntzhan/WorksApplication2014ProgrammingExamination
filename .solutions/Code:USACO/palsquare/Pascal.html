        len1:=0;
        i:=0;
        while ten<>0 do begin
                inc(i);
                if (ten mod b)<10 then num1[i]:=chr((ten mod b)+48)
                else num1[i]:=chr((ten mod b)+55);
                inc(len1);
                ten:=ten div b;
        end;
end;

begin
        len2:=0;
        i:=0;
