#include<cstdio>
#include<cstring>
FILE *fi=fopen("lgame.in","r"),*fd=fopen("lgame.dict","r"),*fo=fopen("lgame.out","w");
const int score[]={2,5,4,4,1,6,5,5,1,7,6,3,5,2,3,5,7,2,1,2,4,6,6,7,5,7};
char a[9],word[2000][9];
int n,cnta[26],cnt[26],value[2000],ans=0,ans1[999],ans2[999],tot;
bool check()
{
	for(int i=0;i<26;++i)
		if(cnt[i]>cnta[i])
			return 0;
	return 1;
}
int main()
{
	int la,i,j,k,l,l2;
	fscanf(fi,"%s",a);
	la=strlen(a);
	for(i=0;i<la;++i)
		++cnta[a[i]-'a'];
	for(n=1;1;++n)
	{
		fscanf(fd,"%s",word[n]);
		if(word[n][0]=='.')
			break;
		l=strlen(word[n]);
		memset(cnt,0,sizeof(cnt));
		bool flag=1;
		for(i=0;i<l;++i)
			++cnt[word[n][i]-'a'];
		for(i=0;i<26;++i)
			if(cnt[i]>cnta[i])
			{
				flag=0;
				--n;
				break;
			}
		if(flag)
		{
			value[n]=0;
			for(i=0;i<26;++i)
				value[n]+=cnt[i]*score[i];
		}
	}
	--n;
	for(i=1;i<=n;++i)
	{
		memset(cnt,0,sizeof(cnt));
		l=strlen(word[i]);
		for(k=0;k<l;++k)
			++cnt[word[i][k]-'a'];
		if(value[i]>ans)
			ans=value[i],ans1[tot=1]=i,ans2[1]=0;
		else if(value[i]==ans)
			ans1[++tot]=i,ans2[tot]=0;
		for(j=i+1;j<=n;++j)
		{
			l2=strlen(word[j]);
			for(k=0;k<l2;++k)
				++cnt[word[j][k]-'a'];
			if(check())
			{
				if(value[i]+value[j]>ans)
					ans=value[i]+value[j],ans1[tot=1]=i,ans2[1]=j;
				else if(value[i]+value[j]==ans)
					ans1[++tot]=i,ans2[tot]=j;
			}
			for(k=0;k<l2;++k)
				--cnt[word[j][k]-'a'];
		}
	}
	fprintf(fo,"%d\n",ans);
	for(i=1;i<=tot;++i)
	{
		fputs(word[ans1[i]],fo);
		if(ans2[i])
			fprintf(fo," %s\n",word[ans2[i]]);
		else
			fputc(10,fo);
	}
	fclose(fi);
	fclose(fo);
	return 0;
}
