inline int
area (window w)
{
    if (w.y1 >= w.y2 || w.x1 >= w.x2) {
	return (0);
    }
    for (int i = 0; i < 256; ++i) {
	if (screen[i].real && screen[i].level > w.level) {
	    if (!
		(w.y2 <= screen[i].y1 || screen[i].y2 <= w.y1
		    || w.x2 <= screen[i].x1 || screen[i].x2 <= w.x1)) {
		window  a (w.y1, w.x1, w.y2, screen[i].x1, w.level);
		window  b (w.y1, screen[i].x2, w.y2, w.x2, w.level);
		window  c (w.y1, max (w.x1, screen[i].x1), screen[i].y1,

		window  d (screen[i].y2, max (w.x1, screen[i].x1), w.y2,
		            min (screen[i].x2, w.x2), w.level);
		return (area (a) + area (b) + area (c) + area (d));

	}
    }
    return ((w.y2 - w.y1) * (w.x2 - w.x1));
}

		l = L[n];
	}
	if(r > R[n]){
		res += dfs(R[n], r, u, d, next[n]);
		r = R[n];
	}
	if(u < U[n]){
		res += dfs(l, r, u, U[n], next[n]);
	}
	if(d > D[n]){
		res += dfs(l, r, D[n], d, next[n]);
	}
	return res;
}
void solve_w(){
	int n = str[2];
	sscanf(str+4, "%d,%d,%d,%d", &L[n], &D[n], &R[n], &U[n]);
	if(R[n] < L[n]) swap(L[n], R[n]);
	if(D[n] < U[n]) swap(U[n], D[n]);
	//cout<<L[n]<<" "<<R[n]<<" "<<U[n]<<" "<<D[n]<<endl;
	addtail(n);
}

	addtail(n);
}

	remove(n);
	addhead(n);
}

	remove(n);
}

	int n = str[2];
	int res = dfs(L[n], R[n], U[n], D[n], next[n]);
	int total = (R[n] - L[n])*(D[n] - U[n]);
	printf("%.3f\n", res/(double)total*100);

}

double query(char ch){
    int hx[50000]={0}, hy[50000]={0};
    int fx[1000], fy[1000], fxn=0, fyn=0;
 
    //step 1, 找出在ch前面的窗口
    //step 2, 离散化 
    for (int i=0; i<256; i++)
        if (window[i].pos != -INF){// && i != ch){
            if (window[i].pos <= window[ch].pos){
                hx[window[i].x1] = true;
                hx[window[i].x2] = true;
                hy[window[i].y1] = true;
                hy[window[i].y2] = true;
            }
        }
    for (int i=0; i<32768; i++){
        if (hx[i]){ fx[fxn++]=i; hx[i]=fxn-1; }//else hx[i]=-1;
        if (hy[i]){ fy[fyn++]=i; hy[i]=fyn-1; }//else hy[i]=-1;
    }
 
    //step 3, fillflood
    int map[300][300] = {0};
    for (int i=0; i<256; i++)
        if (window[i].pos != -INF && i != ch){
            if (window[i].pos < window[ch].pos){
                int bx = hx[window[i].x1],
                    by = hy[window[i].y1];
 
                for (int x=bx; fx[x]<window[i].x2; x++)
                    for (int y=by; fy[y]<window[i].y2; y++)
                        map[x][y] = 1;
            }
        }
 
    //step 4, 计算面积百分比 
    double area = window[ch].area();
    int bx = hx[window[ch].x1],
        by = hy[window[ch].y1];
    double white = 0.0;
    for (int x=bx; fx[x]<window[ch].x2; x++)
        for (int y=by; fy[y]<window[ch].y2; y++){
            if (map[x][y] == 0){
                white += (fx[x+1]-fx[x])*(fy[y+1]-fy[y]);
            }
        }
 
    //cout<<white<<" "<<area<<endl;
    return white/area;
}

double ans(int n, int x1, int y1, int x2, int y2){
	if(!n)
		return (x2 - x1) * (y2 - y1);
	if(w[n].x1 >= x2 || w[n].x2 <= x1 || w[n].y1 >= y2 || w[n].y2 <= y1)
		return ans(w[n].next, x1, y1, x2, y2);
	int tot = 0;
	if(x1 < w[n].x1)
		tot += ans(w[n].next, x1, y1, w[n].x1, y2);
	if(x2 > w[n].x2)
		tot += ans(w[n].next, w[n].x2, y1, x2, y2);
	if(y1 < w[n].y1)
		tot += ans(w[n].next, max(x1, w[n].x1), y1, min(x2, w[n].x2), w[n].y1);
	if(y2 > w[n].y2)

int CutRect(list<char>::iterator i, int x1, int y1, int x2, int y2){ //矩形切割
	if(++i == que.begin() || x1 == x2 || y1 == y2) return (x2 - x1) * (y2 - y1);
	--i;
	int a1 = window[*i]->x1, a2 = window[*i]->x2, b1 = window[*i]->y1, b2 = window[*i]->y2, tot;
	if(!((a1 >= x2) || (a2 <= x1) || (b1 >= y2) || (b2 <= y1))){ //存在交集判断：去掉没有交集的,剩下的就是有交集的
		tot =  CutRect(--i, x1, y1, MIN(a2, x2), MAX(y1, b1)), ++i;
		tot += CutRect(--i, x1, MAX(y1, b1), MAX(x1, a1), y2), ++i;
		tot += CutRect(--i, MAX(x1, a1), MIN(b2, y2), x2, y2), ++i;
		tot += CutRect(--i, MIN(a2, x2), y1, x2, MIN(y2, b2)), ++i;
		return tot;
	}else return CutRect(--i, x1, y1, x2, y2);
}
