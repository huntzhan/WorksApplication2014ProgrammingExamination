    last:array[1..402] of longint;
    flow:array[1..402,1..402] of integer;
    f:array[1..402,1..402] of boolean;
    n,m,max,i,j,k,nn,fence,t,l,r,x,y:longint;
    pcow,pfen:array[1..200] of longint;

    if pcow[i]=0 then begin
