#include <stdio.h>
#define MAX 20000000
 
long n,m;
long map[402][402];
long nam[402];
long num[402];
long las[402];
long ans;
 
long stack[201];
long s,e=1;
 
void bfs(){
	long i;
	for(i=0;i<=n+m+1;i++)nam[i]=MAX;
	stack[0]=n+m+1;
	nam[n+m+1]=0;
	num[0]=1;
	while(s!=e){
		for(i=0;i<=n+m+1;i++)
			if(map[i][stack[s]]!=0 && nam[i]==MAX){
				nam[i]=nam[stack[s]]+1;
				num[nam[i]]++;
				stack[e]=i;
				e++;
			}
		s++;
	}
}
 
void pre(){
	long i,j,k,l;
	scanf("%ld%ld",&n,&m);
	for(i=1;i<=n;i++){
		scanf("%ld",&j);
		for(k=0;k<j;k++){
			scanf("%ld",&l);
			map[i][n+l]=1;
		}
	}
	for(i=1;i<=n;i++)
		map[0][i]=1;
	for(i=1;i<=m;i++)
		map[i+n][n+m+1]=1;
	bfs();
}
 
void add(){
	long i=n+m+1,j=MAX;
	while(i!=0){
		if(j>map[las[i]][i])
			j=map[las[i]][i];
		i=las[i];
	}
	i=n+m+1;
	while(i!=0){
		map[las[i]][i]-=j;
		map[i][las[i]]+=j;
		i=las[i];
	}
	ans+=j;
}
 
void sap(){
	long i=0,j,k;
	while(nam[1]<m){
		k=0;
		if(i==n+m+1){
			add();
			i=0;continue;
		}
		for(j=0;j<=n+m+1;j++)
			if(nam[j]==nam[i]-1 && map[i][j]!=0){
				las[j]=i;
				i=j;
				break;
			}else if(map[i][j]!=0 &&nam[k]>nam[j])k=j;
		if(j==n+m+2){
			num[nam[i]]--;
			if(num[nam[i]]==0)break;
			nam[i]=nam[k]+1;
			if(nam[i]<MAX)num[nam[i]]++;
			if(i!=0)i=las[i];
		}
	}
}
 
main(){
	freopen("stall4.in","r",stdin);
	freopen("stall4.out","w",stdout);
	pre();
	sap();
	printf("%ld\n",ans);
	return 0;
}
