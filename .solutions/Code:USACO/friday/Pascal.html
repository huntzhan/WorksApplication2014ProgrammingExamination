var
    a:array[1..7] of longint;
    n,y,m,d,xq:longint;
begin
    assign(input,'friday.in');reset(input);
    assign(output,'friday.out');rewrite(output);
    readln(n);
    y:=1899;     //设定初始日期，YMD的英文你懂的
    m:=12;
    d:=31;
    xq:=7;      //今天是星期几
    fillchar(a,sizeof(a),0);
    repeat
        inc(d);
        if (m=4) or (m=6) or (m=9) or (m=11) then begin    //改月份
            if d=31 then begin inc(m); d:=1; end;
        end
        else
        if (m<>2) then begin
            if d=32 then begin inc(m); d:=1; end;
        end
        else begin       //对于2月份的特殊处理，晕菜中
            if ((y mod 100>0) and (y mod 4=0)) or ((y mod 100=0) and (y mod 400=0)) then begin
                if d=30 then begin inc(m); d:=1; end;
            end
            else begin
                if d=29 then begin inc(m); d:=1; end;
            end;
        end;
        if m=13 then begin inc(y); m:=1; end;
        inc(xq);
        if xq=8 then xq:=1;
        if d=13 then inc(a[xq]);     //标记
    until (y=1900+n-1) and (m=12) and (d=31);      //到达指定日期
    write(a[6],' ',a[7]);     //输出，特别处理
    for n:=1 to 5 do write(' ',a[n]);    //看清楚，最后没有空格啊啊啊啊！
    writeln;
    close(input);close(output);
end.   //By-LYP
