var
    a:array[0..5001,1..2] of longint;
    n,i,j,k,ans1,ans2:longint;
begin
    assign(input,'milk2.in');reset(input);
    assign(output,'milk2.out');rewrite(output);
    readln(n);
    fillchar(a,sizeof(a),0);
    for i:=1 to n do readln(a[i,1],a[i,2]);
    for i:=1 to n-1 do        //冒泡处理
        for j:=i+1 to n do
            if (a[i,1]>a[j,1]) or ((a[i,1]=a[j,1]) and (a[i,2]>a[j,2])) then begin
                a[0]:=a[i];
                a[i]:=a[j];
                a[j]:=a[0];
            end;
    j:=1;
    repeat
        inc(j);
        if j>n then break;      //写得很猥琐~~
        for i:=1 to j-1 do
            if a[j,1]<=a[i,2] then begin      //判断线段能否合并
                if a[j,2]>a[i,2] then a[i,2]:=a[j,2];
                for k:=j to n do a[k]:=a[k+1];        //删除它，其实可以用标记，小菜时写的，太猥琐啊
                dec(n);
                dec(j);
                break;
            end;
    until false;
    ans1:=0;
    ans2:=0;
    for i:=1 to n do begin
        if a[i,2]-a[i,1]>ans1 then ans1:=a[i,2]-a[i,1];       //统计答案
        if a[i+1,1]-a[i,2]>ans2 then ans2:=a[i+1,1]-a[i,2];
    end;
    writeln(ans1,' ',ans2);
    close(input);close(output);
end. //By-LYP

var i:integer;
begin
    for i:=1 to 2*n do
        if a=line[i] then
        begin
            search:=i;
            exit;
        end;
end;
 
procedure qsort(a,b:integer);
var l,r,mid,temp:longint;
begin
    if a=b then exit;
    l:=a;
    r:=b;
    mid:=line[(l+r)div 2];
    repeat
        while line[l]<mid do l:=l+1;
        while line[r]>mid do r:=r-1;
        if l<=r then
        begin
            temp:=line[l];
            line[l]:=line[r];
            line[r]:=temp;
            l:=l+1;
            r:=r-1;
        end;
    until l>r;
    if a<r then qsort(a,r);
    if l<b then qsort(l,b);
end;

procedure qsort(l,r:longint);
  var i,j,m,t:longint;
  begin
    i:=l; j:=r; m:=a[(l+r+r) div 3];
    while i<j do
      begin
        while a[i]<m do inc(i);
        while a[j]>m do dec(j);
        if i<=j then
          begin
            t:=a[i]; a[i]:=a[j]; a[j]:=t;
            t:=b[i]; b[i]:=b[j]; b[j]:=t; 
            inc(i); dec(j);
          end;
      end;
    if i<r then qsort(i,r);
    if l<j then qsort(l,j);
  end;
begin
assign(input,'milk2.in');

   read(n);

   for i:=1 to n do                             //预处理
    begin
     read(x,y);

    x:=0;
    j:=0;

     x:=x+a[i];                                 //统计有无人分界线

procedure qsort(l,r,p:longint);//快排……
var
    i,j,mid:longint;
begin
    i:=l;
    j:=r;
    mid:=a[(l+r) shr 1,p];
    repeat
        while a[i,p]<mid do inc(i);
        while a[j,p]>mid do dec(j);
        if i<=j then
        begin
            a[0]:=a[i];
            a[i]:=a[j];
            a[j]:=a[0];
            inc(i);dec(j);
        end;
    until i>j;
    if l<j then qsort(l,j,p);
    if i<r then qsort(i,r,p);
end;
