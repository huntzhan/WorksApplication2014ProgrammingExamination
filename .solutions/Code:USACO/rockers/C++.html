int n,t,m,tim[20],ton[21],mmin=0;
 
void greedy()//贪心
{
	int i,j,cont=0,k=m,a[20];
	for(i=0;i<n;i++) a[i]=tim[i];
	if(k>n) k=n;
	for(i=0;i<k;i++)
	{
		int s=t;
		sort(a+ton[i],a+ton[i+1]);
		for(j=ton[i];j<ton[i+1];j++)
			if(s>=a[j]) s-=a[j],cont++;
			else break;
	}
	if(mmin<cont) mmin=cont;
}
 
void dfs(int d,int t)//枚举位置
{
	if(d==m || d==n) {ton[d]=n; greedy(); return;}
	if(t>=n) return;
	for(int i=t;i<n;i++)
	{
		ton[d]=i;
		dfs(d+1,i+1);
	}
}
