VAR
  F:ARRAY [0..20,0..20,0..20] OF INTEGER;
  A:ARRAY [1..20] OF INTEGER;
  I,M,V,N,J,K:LONGINT;
 
PROCEDURE INIT;
BEGIN
  ASSIGN(INPUT,'rockers.in');
  RESET(INPUT);
  READLN(N,V,M);
  FOR I:= 1 TO N DO
    READ(A[I]);
  CLOSE(INPUT);
  FILLCHAR(F,SIZEOF(F),0);
END;
 
FUNCTION MAX(X,Y:LONGINT):LONGINT;
BEGIN
  IF X>Y THEN EXIT(X)
  ELSE EXIT(Y);
END;
 
PROCEDURE MAIN;
BEGIN
  FOR I:= 1 TO N DO
    FOR J:= 1 TO M DO
      FOR K:= 0 TO V DO BEGIN
        IF K<A[I] THEN
          F[I,J,K]:=MAX(F[I,J-1,V],F[I-1,J,K])
        ELSE
          F[I,J,K]:=MAX(F[I-1,J-1,V]+1,F[I-1,J,K-A[I]]+1);
      END;
END;

  INIT;

  REWRITE(OUTPUT);
  WRITELN(F[N,M,V]);
