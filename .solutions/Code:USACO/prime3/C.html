void sort_ans(long s,long t)
{
	long l = s,r = t;
	matrix key = ans[(t + s) >> 1];
 
	while (l <= r)
	{
		while (comp(ans[l],key)) l++;
		while (comp(key,ans[r])) r--;
		if (l <= r)
		{
			swap(ans[l],ans[r]);
			l++;
			r--;
		}
	}
 
	if (s < r) sort_ans(s,r);
	if (l < t) sort_ans(l,t);
}
