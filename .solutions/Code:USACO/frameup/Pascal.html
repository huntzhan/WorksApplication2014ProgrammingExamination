Procedure Tsort(Deep	   : Longint);
Var
   I   : Longint;
   C,J : Char;
Begin
   For C:='A' To 'Z' Do
      If (C In S)And(Ini[C]=0) Then
      Begin
	 Sort:=Sort+C;
	 For I:=1 To Ira[C] Do Dec(Ini[A[C,I]]);
	 If Deep<N Then
	 Begin
	    S:=S-[C];
	    Tsort(Deep+1);
	    S:=S+[C];
	 End
	 Else
	 Begin
	    Inc(Lamda);
	    Ans[Lamda]:=Sort;
	 End;
	 Delete(Sort,Length(Sort),1);
	 For I:=1 To Ira[C] Do Inc(Ini[A[C,I]]);
      End;
End; { Tsort }

	    Delta[M[I,J]]:=True;

	    If (I<Lly[M[I,J]])Or(Lly[M[I,J]]=0) Then Lly[M[I,J]]:=I;
	    If J>Urx[M[I,J]] Then Urx[M[I,J]]:=J;
	    If I>Ury[M[I,J]] Then Ury[M[I,J]]:=I;
	 End;
      End;

	       M1[C,M[Lly[C],I]]:=True;

	    If (M[Ury[C],I]<>C)And(Not(M1[C,M[Ury[C],I]])) Then
	    Begin
	       M1[C,M[Ury[C],I]]:=True;
	       Inc(Ira[C]);

	    End;
	 End;
	 For I:=Lly[C] To Ury[C] Do

	    Begin
	       M1[C,M[I,Llx[C]]]:=True;

	       M1[C,M[I,Urx[C]]]:=True;

	    End;	    
	 End;
      End;
   For C:='A' To 'Z' Do If Delta[C] Then Begin N:=N+1;S:=S+[C];End;
   For C:='A' To 'Z' Do

   Tsort(1);

   Close(Output);

procedure qsort(l,r:integer);//出解时排序
var i,j:integer; mid:string;
begin
    i:=l;j:=r; mid:=ans[(i+j) shr 1];
	repeat
		while ans[i]<mid do inc(i);
		while ans[j]>mid do dec(j);
		if i<=j then
			begin
				st:=ans[i]; ans[i]:=ans[j]; ans[j]:=st; inc(i); dec(j);
			end;
	until i>j;
	if i<r then qsort(i,r); if l<j then qsort(l,j);
end;

  procedure del(x:char);//处理当前矩形（hash值加1）
var i,x1,x2,y1,y2,k:integer;
begin
k:=ord(x)-64;
x1:=p[k,1,1];x2:=p[k,1,2]; y1:=p[k,2,1];y2:=p[k,2,2];
for i:=x1 to x2 do inc(hash[i,y1]);
for i:=x1 to x2 do inc(hash[i,y2]);
for i:=y1 to y2 do inc(hash[x1,i]);
for i:=y1 to y2 do inc(hash[x2,i]);
exi[k]:=false;
end;
	procedure recover(x:char);//深搜还原（hash减1）
var i,x1,x2,y1,y2,k:integer;
begin
k:=ord(x)-64;
x1:=p[k,1,1];x2:=p[k,1,2]; y1:=p[k,2,1];y2:=p[k,2,2];
for i:=x1 to x2 do dec(hash[i,y1]);
for i:=x1 to x2 do dec(hash[i,y2]);
for i:=y1 to y2 do dec(hash[x1,i]);
for i:=y1 to y2 do dec(hash[x2,i]);
exi[k]:=true;
end;
  function pan(x:char):boolean;//判断当前矩形是否在表层
