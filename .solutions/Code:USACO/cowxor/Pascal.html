procedure       trie_search ;
var     tmp , p , step
                : longint ;
begin
        p := 0 ;
        for step := 20 downto 0 do begin
         tmp := ord (u and (1 shl step) > 0) ;
         if (t[p].next[tmp xor 1] <> 0) then begin
          inc (now , (1 shl step)) ;
          p := t[p].next[tmp xor 1] ;
         end
         else
          p := t[p].next[tmp] ;
        end ;
 
        if (now > ans) then begin
         ans := now ;
         st := t[p].d ;
         en := ci ;
        end ;
end ;
 
procedure       trie_insert ;
var     p , i , tmp
                : longint ;
begin
        p := 0 ;

         tmp := ord (u and (1 shl i) > 0) ;

          inc (tot) ;
          t[p].next[tmp] := tot ;
          p := tot ;
         end

          p := t[p].next[tmp] ;

var
a:array[0..4200000]of boolean;//判断以i为前缀的是否出现
b:array[0..100001]of longint;
ans,start,ends,n,x,i,j,k,l:longint;
begin
  assign(input,'cowxor.in');reset(input);
  assign(output,'cowxor.out');rewrite(output);
  read(n);
  read(b[1]);
  for i:=2 to n do
  begin
        read(x);
        b[i]:=b[i-1] xor x;
  end;
  for i:=1 to 21 do
  a[1 shl i]:=true;//初始0,若前缀为a1a2a3a4...ai,记得在树组a中的位置为1a1a2a3a4...ai(在最前面加个'1')
  ans:=b[1];
  start:=1;ends:=1;
  for i:=1 to n do
  begin
        x:=1;
        for j:=1 to 21 do
        begin
                if a[(x shl 1+((b[i] shr (21-j))and 1)xor 1)]then
                x:=x shl 1+((b[i] shr (21-j))and 1)xor 1 else
                x:=x shl 1+(b[i] shr (21-j))and 1;  //已处理完前j-1位,判断第j位是b[i]的第j位的异或的前缀是否存在
        end;
        if b[i] xor (x xor (1 shl 21))>ans then  //在判断大小时记得把类似符号位的'1'去掉
        begin
                ans:=b[i] xor (x xor (1 shl 21));
                ends:=i;
        end;
        x:=b[i]+1 shl 21;  //第i个数的前缀处理,不加'1'会出错,如0001和0011,不加'1',前者的位置是1,后者前缀的前三位位置也为1,
                           //加'1',前者为'10001',后者为'1001'
        for j:=21 downto 1 do
        begin
                a[x]:=true;
                x:=x shr 1;
        end;
  end;
  for start:=ends downto 1 do if b[start-1] xor b[ends]=ans then break;
  writeln(ans,' ',start,' ',ends);
close(input);
close(output);
end.

program cowxor;//By Fenix.L
const mh=4194304;//2^22
      mn=100000;
var
  tr:array [1..mh] of boolean;//trie数组
  w,v,dig:array [0..mn] of longint;//w:牛的值 v[i]:牛1到牛i的xor值 dig[i]:储存2^(i-1)，方便观察
  n:longint;
 
  procedure inout;
  begin
    assign(input,'cowxor.in');
    reset(input);
    assign(output,'cowxor.out');
    rewrite(output);
  end;
 
  procedure cio;
  begin
    close(input);
    close(output);
  end;
 
  procedure init;
  var a:longint;
  begin
    readln(n);
    for a:=1 to n do readln(w[a]);
 
    v[1]:=w[1];
    for a:=2 to n do v[a]:=w[a] xor v[a-1];
  end;
 
  procedure work;
  var ans,anss,anst,i,j,x,a,now:longint;//ans:全局最优解 now:当前最优解
  begin
    ans:=0; anst:=1; v[0]:=0;
    fillchar(tr,sizeof(tr),false);
    dig[21]:=1;
    for a:=20 downto 1 do begin
        dig[a]:=dig[a+1] shl 1;
        tr[dig[a]]:=true;
    end;
    tr[dig[1] shl 1]:=true;//初始化trie dig数组
 
    for a:=1 to n do begin
        i:=1; j:=1; now:=0;//i:在trie中位置 j:循环到牛a的第几位
        while j<=21 do begin
              x:=v[a] and dig[j] shr (21-j);//提取牛a的第j位
              if tr[i+i+(1-x)] then begin
                 i:=i+i+(1-x);
                 now:=now+dig[j];
              end
              else i:=i+i+x;
              inc(j);
        end;
        if now>ans then begin ans:=now; anst:=a; end;
 
        i:=1; j:=1;
        while j<=21 do begin
              i:=i+i+v[a] and dig[j] shr (21-j);
              tr[i]:=true;
              inc(j);
        end;
    end;
 
    for a:=anst-1 downto 0 do if v[a] xor v[anst]=ans then begin
        anss:=a;
        break;
    end;
 
    writeln(ans,' ',anss+1,' ',anst);
  end;
 
begin
  inout;
  init;
  work;
  cio;
