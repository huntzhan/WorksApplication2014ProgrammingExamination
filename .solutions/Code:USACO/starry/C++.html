			t.x+=walk[j][0],t.y+=walk[j][1];
			if(c[t.x][t.y]=='1' && !mark[t.x][t.y] && ok(t.x,t.y))
				p[++np]=t,mark[t.x][t.y]=1;
		}
	for(i=1;i<np;++i)
		for(j=i+1;j<=np;++j)
			ans+=sqrt(pf(p[i].x-p[j].x)+pf(p[i].y-p[j].y));
	return ans;
}
