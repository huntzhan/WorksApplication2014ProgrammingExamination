char *movestr[] = { "abde", "abc", "bcef", "adg", "bdefh", "cfi", "degh",
                    "ghi", "efhi" };
 
int movedist[9][9];
int clock[9];
 
int bestmove[9];
int nbestmove;
 
/* translate move strings into array "movedist" of which clocks change on each move */
void
mkmove(void)
{
    int i;
    char *p;
 
    for(i=0; i<9; i++)
	for(p=movestr[i]; *p; p++)
	    movedist[i][*p-'a'] = 3;
}
 
/* apply some number of each move from k to 9 */
/* move contains the number of times each move is applied */
void
solve(int *move, int k)
{
    int i, j, n, rep;
 
    if(k == 9) {
	for(j=0; j<9; j++)
	    if(clock[j]%12 != 0)
		return;
 
	/* we have a successful sequence of moves */
	n = 0;
	for(j=0; j<9; j++)
	    n += move[j];
 
	if(nbestmove == 0 || n < nbestmove) {
	    nbestmove = n;
	    for(j=0; j<9; j++)
		bestmove[j] = move[j];
	}
	return;
    }
 
    /*
     * the for loop counts down so we
     * generate smaller numbers first by
     * trying more of small numbered
     * moves before we try less of them.
     */
    for(rep=3; rep>=0; rep--) {
	/* apply move k rep times */
	for(i=0; i<rep; i++)
	    for(j=0; j<9; j++)
		clock[j] += movedist[k][j];
	move[k] = rep;
 
	solve(move, k+1);
 
	/* undo move */
	for(i=0; i<rep; i++)
	    for(j=0; j<9; j++)
		clock[j] -= movedist[k][j];
    }
}
 
void
main(void)
{
    FILE *fin, *fout;
    int i, j, move[9];
    char *sep;
 
    fin = fopen("clocks.in", "r");
    fout = fopen("clocks.out", "w");
    assert(fin != NULL && fout != NULL);
 
    mkmove();
 
    for(i=0; i<9; i++)
	fscanf(fin, "%d", &clock[i]);
 
    solve(move, 0);
 
    sep = "";
    for(i=0; i<9; i++) {
	for(j=0; j<bestmove[i]; j++) {
	    fprintf(fout, "%s%d", sep, i+1);
	    sep = " ";
	}
    }
    fprintf(fout, "\n");
 
    exit(0);
}

int invert()
{
    int sum,i,j,k;
    k=1;
    sum=0;
    for(i=0;i<9;i++)
    {
        sum+=clock[i]*k;
        k*=4;
    }
    return sum;
}

    int i,j,k;

    int i,j,k;

{
 
    int i,j,k,s,t,temp[9];
    fp=fopen("clocks.in","r");
    fq=fopen("clocks.out","w");
    memset(flag,-1,sizeof(flag));
    memset(road,-1,sizeof(road));
    for(i=0;i<9;i++)

        clock[i]=(clock[i]/3)%4;

    k=invert();
    queue[rear++]=k;
    flag[k]=1;
    while(front!=rear)
    {
        k=queue[front++];
        change(k);
        for(j=0;j<9;j++)
        temp[j]=clock[j];
        for(i=1;i<=9;i++)
        {
            act(i);
            s=invert();
            if(flag[s]==-1)
            {
                flag[s]=1;
                queue[rear++]=s;
                road[s][0]=k;
                road[s][1]=i;
            }
            for(j=0;j<9;j++)
            clock[j]=temp[j];
        }
        if(flag[0]==1)
        break;

    print(0);
    for(i=0;i<top-1;i++)
    fprintf(fq,"%d ",stack[i]);

    fclose(fp);

    return 0;
}

void changeform(int K, int N)
{
     int i;
     for(i=1; i<=N; ++i)
     {
              switch(K)
              {
                  case 1: add('A'); add('B'); add('D'); add('E'); break;
                  case 2: add('A'); add('B'); add('C'); break;
                  case 3: add('B'); add('C'); add('E'); add('F'); break;
                  case 4: add('A'); add('D'); add('G'); break;
                  case 5: add('B'); add('D'); add('E'); add('F'); add('H'); break;
                  case 6: add('C'); add('F'); add('I'); break;
                  case 7: add('D'); add('E'); add('G'); add('H'); break;
                  case 8: add('G'); add('H'); add('I'); break;
                  case 9: add('E'); add('F'); add('H'); add('I'); break;
              }
     }
}
 
void undoform(int K, int N)
{
     int i;
     for(i=1; i<=N; ++i)
     {
              switch(K)
              {
                  case 1: undo('A'); undo('B'); undo('D'); undo('E'); break;
                  case 2: undo('A'); undo('B'); undo('C'); break;
                  case 3: undo('B'); undo('C'); undo('E'); undo('F'); break;
                  case 4: undo('A'); undo('D'); undo('G'); break;
                  case 5: undo('B'); undo('D'); undo('E'); undo('F'); undo('H'); break;
                  case 6: undo('C'); undo('F'); undo('I'); break;
                  case 7: undo('D'); undo('E'); undo('G'); undo('H'); break;
                  case 8: undo('G'); undo('H'); undo('I'); break;
                  case 9: undo('E'); undo('F'); undo('H'); undo('I'); break;
              }
     }
}
 
void add(char ch)
{
     int obj;
     obj=ch-'A'+1;
     clocks[obj]+=3;
     if(clocks[obj]>12) clocks[obj]-=12;
}
 
void undo(char ch)
{
     int obj;
     obj=ch-'A'+1;
     clocks[obj]-=3;
     if(clocks[obj]<3) clocks[obj]=12;
}
