// stably sort a[0..n-1] to b[0..n-1] with keys in 0..K from r
static void radixPass(int* a, int* b, int* r, int n, int K)
{
	// count occurrences
	int* c = new int[K + 1]; // counter array
	for (int i = 0; i <= K; i++) c[i] = 0; // reset counters
	for (int i = 0; i < n; i++) c[r[a[i]]]++; // count occurrences
	for (int i = 0, sum = 0; i <= K; i++) // exclusive prefix sums
	{
		int t = c[i]; c[i] = sum; sum += t;
	}
	for (int i = 0; i < n; i++) b[c[r[a[i]]]++] = a[i]; // sort
	delete [] c;
}
 
// find the suffix array SA of T[0..n-1] in {1..K}^n
// require T[n]=T[n+1]=T[n+2]=0, n>=2
void suffixArray(int* T, int* SA, int n, int K) {
 
	int n0=(n+2)/3, n1=(n+1)/3, n2=n/3, n02=n0+n2;
	int* R = new int[n02 + 3]; R[n02]= R[n02+1]= R[n02+2]=0;
	int* SA12 = new int[n02 + 3]; SA12[n02]=SA12[n02+1]=SA12[n02+2]=0;
	int* R0 = new int[n0];
	int* SA0 = new int[n0];
 
	//******* Step 0: Construct sample ********
	// generate positions of mod 1 and mod 2 suffixes
	// the "+(n0-n1)" adds a dummy mod 1 suffix if n%3 == 1
	for (int i=0, j=0; i < n+(n0-n1); i++) if (i%3 != 0) R[j++] = i;
 
	//******* Step 1: Sort sample suffixes ********
	// lsb radix sort the mod 1 and mod 2 triples
	radixPass(R , SA12, T+2, n02, K);
	radixPass(SA12, R , T+1, n02, K);
	radixPass(R , SA12, T , n02, K);
	// find lexicographic names of triples and
	// write them to correct places in R
 
	int name = 0, c0 = -1, c1 = -1, c2 = -1;
	for (int i = 0; i < n02; i++) {
		if (T[SA12[i]] != c0 || T[SA12[i]+1] != c1 || T[SA12[i]+2] != c2)
		{
			name++; c0 = T[SA12[i]]; c1 = T[SA12[i]+1]; c2 = T[SA12[i]+2];
		}
		if (SA12[i] % 3 == 1) { R[SA12[i]/3] = name; } // write to R1
		else {
			R[SA12[i]/3 + n0] = name;
		} // write to R2
	}
	// recurse if names are not yet unique
	if (name < n02) {
		suffixArray(R, SA12, n02, name);
		// store unique names in R using the suffix array
		for (int i = 0; i < n02; i++) R[SA12[i]] = i + 1;
	} else // generate the suffix array of R directly
		for (int i = 0; i < n02; i++) SA12[R[i] - 1] = i;
 
	//******* Step 2: Sort nonsample suffixes ********
	// stably sort the mod 0 suffixes from SA12 by their first character
	for (int i=0, j=0; i < n02; i++) if (SA12[i] < n0) R0[j++] = 3*SA12[i];
	radixPass(R0, SA0, T, n0, K);
 
	//******* Step 3: Merge ********
	// merge sorted SA0 suffixes and sorted SA12 suffixes
	for (int p=0, t=n0-n1, k=0; k < n; k++) {
#define GetI() (SA12[t] < n0 ? SA12[t] * 3 + 1 : (SA12[t] - n0) * 3 + 2)
		int i = GetI(); // pos of current offset 12 suffix
		int j = SA0[p]; // pos of current offset 0 suffix
		if (SA12[t] < n0 ? // different compares for mod 1 and mod 2 suffixes
			leq(T[i], R[SA12[t] + n0], T[j], R[j/3]) :
		leq(T[i],T[i+1],R[SA12[t]-n0+1], T[j],T[j+1],R[j/3+n0]))
		{ // suffix from SA12 is smaller
			SA[k] = i; t++;
			if (t == n02) // done --- only SA0 suffixes left
				for (k++; p < n0; p++, k++) SA[k] = SA0[p];
		} else { // suffix from SA0 is smaller
			SA[k] = j; p++;
			if (p == n0) // done --- only SA12 suffixes left
				for (k++; t < n02; t++, k++) SA[k] = GetI();
		}
	}
	delete [] R; delete [] SA12; delete [] SA0; delete [] R0;
}

void extend_kmp(char* b,int *nextb)
{
	int i=0,j,k;
	while(b[i]==b[i+1])++i;
	nextb[1]=i;
	k=1;
	for(i=2;b[i];++i)
	{
		if(nextb[i-k]+i<=nextb[k]+k)
			nextb[i]=nextb[i-k];
		else
		{
			j=nextb[k]+k-i;
			if(j<0)j=0;
			while(b[j]==b[j+i]&&b[j+i])++j;
			nextb[i]=j;
		}
	}
}
 
int match(char* a,char* b,int *nexta,int *nextb,int len)
{
	extend_kmp(b,nextb);
	int i,j,k;
	i=0,j=0,k=0;
	while(a[i]==b[i])++i;
	nexta[0]=i;
	for(i=1;i<len;++i)
	{
		if(nextb[i-k]+i<nexta[k]+k)
			nexta[i]=nextb[i-k];
		else 
		{
			j=nexta[k]+k-i;
			if(j<0)j=0;
			while(a[j+i]==b[j]&&j+i<len&&b[j])
				++j;
			nexta[i]=j;
			k=i;
		}
	}
}
 
void rev(char *a,int len)
{
	char t;
	for(int i=0;i<len/2;++i)
		t=a[i],a[i]=a[len-i-1],a[len-i-1]=t;
}
 
void solve(char* a,int len,int first)
{
	if(len<=1)return;
	int mid=len>>1;
	int i,j,k;
 
	for(i=mid;i<len;++i)b[i-mid]=a[i];
	b[i-mid]=0;
	rev(a,len);
	match(a,b,nexta1,nextb,len);
	rev(a,len);
	for(i=0;i<mid;++i)b[i]=a[mid-i-1];
	b[i]=0;
	match(a,b,nexta2,nextb,len);
	nexta1[len]=nexta2[len]=0;
 
	for(i=mid;i<len;++i)
		if(nexta1[len-i]>=(i-mid)/2)
		{
			int x=(i-mid)+2*nexta2[i];
			if((x>=answer&&first+mid-nexta2[i]<start)||x>answer)
			{
				answer=x;
				start=first+mid-nexta2[i];
				end=first+i+nexta2[i]-1;
				place=start;
			}
		}
 
	for(i=0;i<mid;++i)
		if(nexta2[i]>=(mid-i)/2)
		{
			int x=(mid-i)+2*nexta1[len-i];
			if((x>=answer&&first+i-nexta1[len-i]<start)||x>answer)
			{
				start=first+i-nexta1[len-i];
				end=first+mid+nexta1[len-i]-1;
				place=start;
				answer=x;
			}
		}
 
	solve(a,mid-1,first);
	solve(a+mid,len-mid,first+mid);
}

#include <iostream>
#include <fstream>
 
using namespace std;
 
ifstream fin("calfflac.in");
ofstream fout("calfflac.out");
 
int main()
{
    int i = 1, j, length, max_length = 0, ori_location, record;
    char str[20000],cl_str[20000];
    while (!fin.eof()) str[i++] = fin.get();
    length = i;
    for (i = 1; i<length; i++)
    {
        if (!((str[i] >= 'a' && str[i] <= 'z') || (str[i] >= 'A' && str[i] <= 'Z'))) cl_str[i] = ' ';
        else if(str[i] >= 'a' && str[i] <= 'z') cl_str[i] = str[i] - 'a' + 'A';
        else cl_str[i] = str[i];
    }
    for (i = 1; i < length; i++)
    {
        int a_blank = 0, b_blank = 0;
        for(int flag=1;flag >= 0;flag--)
        {
            if (flag == 0 && cl_str[i] == ' ') break;
            for (j = 1; i-j-a_blank >= 0 && i+j+b_blank-flag < length; j++)
            {
                while (i-j-a_blank>=0 && cl_str[i-j-a_blank] == ' ') a_blank++;
                while (i+j+b_blank-flag<length && cl_str[i+j+b_blank-flag] == ' ') b_blank++;
                if (cl_str[i-j-a_blank] != cl_str[i+j+b_blank-flag])
                {
                    int temp = 2 * (--j) + 1 - flag;
                    if (temp > max_length)
                    {
                        max_length = temp;
                        while(cl_str[i-j-a_blank] == ' ') a_blank--;
                        while(cl_str[i+j+b_blank-flag] == ' ') b_blank--;
                        ori_location = i-j-a_blank;
                        record = 2*j+1-flag+a_blank+b_blank;
                    }
                    a_blank = 0, b_blank = 0;
                    break;
                }
            }
        }
    }
    fout<<max_length<<endl;
    for(i=ori_location;i<ori_location + record;i++) fout<<str[i];
    fout<<endl;
    fin.close();
    fout.close();
    return 0;
}
