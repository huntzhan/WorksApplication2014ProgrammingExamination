    double c = dis(p.bx, p.by, p.ex, p.ey);
    if(!dblcmp(p.bx - p.ex)){
        if(dblcmp((y - p.ey) * (y - p.by)) <= 0)
            return fabs(x - p.bx);
        else return min(a, b);
    }
    if(!dblcmp(p.by - p.ey)){
        if(dblcmp((x - p.ex) * (x - p.bx)) <= 0)
            return fabs(y - p.by);
        else return min(a, b);
    }
    if(dblcmp(a * a + c * c - b * b) >= 0 && dblcmp(b * b + c * c - a * a) >= 0){
        return a * sin(acos((a * a + c * c - b * b) / (2 * a * c)));
    }else return min(a, b);
}
double SumDis(double x, double y)
{
    double sum = 0;
    for(int i = 0; i < n; i++)
        sum += PoiToSeg(x, y, p[i]);
    return sum;
}

double dist(NODE& q,NODE& s,NODE& t)
{
	double a=dist(q,s);
	if(a<MIN)return 0;
	double b=dist(q,t);
	if(b<MIN)return 0;
	double c=dist(s,t);
	if(c<MIN)return a;
	if(a*a>=b*b+c*c) 
  		return b;
 	if(b*b>=a*a+c*c)  
  		return a;
	float l=(a+b+c)/2;
 	float S=sqrt(l*(l-a)*(l-b)*(l-c));
 	return 2*S/c;     
}

double totdist(const double &x, const double &y){
	double tot  = 0;
	for(int i = 1; i <= F; i++){
		if(Node[i].y1 == Node[i].y2){ // 平行于x
			if(x >= Node[i].x1 && x <= Node[i].x2) // 可以做垂线
				tot += fabs(y - Node[i].y1);
			else if(x > Node[i].x2)
				tot += dist(x, y, Node[i].x2, Node[i].y2);
			else
				tot += dist(x, y, Node[i].x1, Node[i].y1);
		}else{	// 与y平行

inline void anneal(fence map, int now){
	for(int i = map.x1 - step[now - 1]; i <= map.x2 + step[now - 1]; i += step[now]){
		for(int j = map.y1 - step[now - 1]; j <= map.y2 + step[now - 1]; j += step[now]){
			double count = totdist(i, j);
			if (l > count){
				l = count;
				x = i;
				y = j;
			}
		}
	}
}
