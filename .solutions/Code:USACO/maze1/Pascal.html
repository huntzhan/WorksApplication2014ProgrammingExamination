  var i,x1,y1:byte;
  begin
    a[x,y]:=false;
    for i:=1 to 4 do begin
      x1:=x+p[i,1];y1:=y+p[i,2];

        x1:=x1+p[i,1];y1:=y1+p[i,2];

          b[x1,y1]:=b[x,y]+1;work(x1,y1);
        end;
      end;
    end;

  end;

var tt,uu:integer;
begin
  max:=order+1;
  flag:=false;
  for tt:=0 to 2*n+2 do
  for uu:=0 to 2*m+2 do
    if map[tt][uu]=order then
    begin
      if (map[tt][uu+1]<>-2) and (map[tt][uu+2]=-1) then

      if (map[tt-1][uu]<>-2) and (map[tt-2][uu]=-1) then
        begin map[tt-2][uu]:=max; flag:=true; end;
    end;
  if flag then find(max);
end;

      procedure init;
        var
          i,j:longint;
          c:string;
        begin
          assign(input,'maze1.in');
          assign(output,'maze1.out');
          reset(input);
          rewrite(output);

          fillchar(w,sizeof(w),false);
          x0:=0;
          for i:=1 to n*2+1 do
            begin
              readln(c);
              for j:=1 to m*2+1-length(c) do c:=c+' ';
              for j:=1 to length(c) do
                if c[j]=' ' then
                  begin
                    w[i][j]:=true;

                            x0:=i;
                            y0:=j;
                          end

                          x1:=i;
                          y1:=j;
                        end;
                      end;
                  end;
            end;
