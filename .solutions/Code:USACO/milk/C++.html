using namespace std;
class farmer
{
    public:
    //p=price,a=amount
        int p,a;
        bool operator < (const farmer &b) const
        {
            return p > b.p;
        }
};
int Greedy(priority_queue<farmer> *F,int m,farmer *f)
{
    int v=0;
    while(0<m)
    {
        *f=F->top();
        F->pop();
        if(f->a <=m)
        {
            v+=f->a * f->p;
            m-=f->a;
        }
        else
        {
            v+=f->p * m;
            m=0;
        }
    }
    return v;
}
int main()
{
    ifstream cin("milk.in");
    ofstream cout("milk.out");
    priority_queue<farmer> F;
    farmer *f=new farmer;
    int m,n;
    cin >> m >> n;
    for(int i=0;i<n;i++)
    {
        cin >> f->p >> f->a;
        F.push(*f);
    }
    int k=Greedy(&F,m,f);
    cout << k << endl ;
    return 0;
}

int qsort(int l, int r)
{
    int i, j, k;
    i = l; j = r; k = a[rand() % (r - l + 1) + l];
    do
    {
      while (i <= j && a[i] < k) i++;
      while (i <= j && a[j] > k) j--;
      if (i <= j)
      {
         a[0] = a[i]; a[i] = a[j]; a[j] = a[0];
         b[0] = b[i]; b[i] = b[j]; b[j] = b[0];
         i++; j--;
      }
    } while (i <= j);
    if (l < j) qsort(l, j);
    if (i < r) qsort(i, r);
}
