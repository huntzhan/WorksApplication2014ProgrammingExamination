void dfs(int row, int ld, int rd, int i)
{
	int p, pos;
	if(row!=aim)
	{
		pos=aim&(~(row|ld|rd));
		while(pos!=0)
		{
			p=pos&(pos^(pos-1));
			pos-=p;
			if(ans<3)
			{
				int l=1, k=p;
				while(k!=1)
				{
					k=k>>1;
					l++;
				}
				a[i]=l;
			}
			dfs(row|p,(ld|p)<<1,(rd|p)>>1,i+1);
		}
	}
	else
	{
		ans++;
		if(ans<=3)
		{
			printf("%d", a[1]);
			for(int k=2; k<=n; k++)
				printf(" %d", a[k]);
			printf("\n");	
		}
	}
}

void DFS(int k, int column, int lcorner, int rcorner)
{
	if (k == n)
	{
		if (++res <= 3)
		{
			printf("%d", a[0]+1);
			for (int j = 1; j < n; ++j)
				printf(" %d", a[j]+1);
			putchar('\n');
		}

	}
	for (int p = full & ~(column | lcorner | rcorner); p; )
	{
		int i = p & -p;
		p -= i;
		if (res < 3) a[k] = log(double(i+0.5))/log(2.);
		DFS(k+1, column+i, (lcorner+i)/2, (rcorner+i)*2);

}

{
    int pos(0), p(0);
    if (row != a)
    {
        pos = a & ~(l | r | row);
