int INSERT(int s,int t,int f,int rank){
	node *p=new(node);
	(*p).t=t;
	(*p).f=f;
	(*p).rank=rank;
	(*p).next=head[s];
	map[s][t]=p;
	head[s]=p;
	return 0;
}
 
int ADD(int s,int t,int f){
	(*map[s][t]).f+=f;
	return 0;
}
 
int sap(int v,int flow){
	int rec=0;
	if (v==n){
		return flow;
	}
	node *p=d[v];
	while (p!=NULL){
		if ((*p).f&&h[v]==h[(*p).t]+1){
			int ret=sap((*p).t,min(flow-rec,(*p).f));
			(*p).f-=ret;
			ADD((*p).t,v,ret);
			d[v]=p;
			if ((rec+=ret)==flow){
				return flow;
			}
		}
		p=(*p).next;
	}
	if (!(--gap[h[v]])){
		h[1]=n;
	}
	gap[++h[v]]++;
	d[v]=head[v];
	return rec;
}
 
int maxflow(){
	int rec=0;
	memset(gap,0,sizeof(gap));
	memset(h,0,sizeof(h));
	for (int i=0;i++<n;){
		d[i]=head[i];
	}
	gap[0]=n;
	while (h[1]<n){
		rec+=sap(1,0x7fffffff);
	}
	return rec;
}
