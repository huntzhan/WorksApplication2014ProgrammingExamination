  t1,t2,t3:vec3;
begin
  for i:=0 to 99 do
  begin
    t1:=xx[1]*i;
    for j:=0 to 99 do
    begin
      t2:=t1+xx[2]*j;
      for k:=0 to 99 do if i+j+k>0 then
      begin
        t3[1]:=t2[1]+xx[3,1]*k;
        if t3[1]<des[1] then continue;
        t3[2]:=t2[2]+xx[3,2]*k;
        if (t3[2]<des[2])or(t3[1]*des[2]<>t3[2]*des[1]) then continue;
        t3[3]:=t2[3]+xx[3,3]*k;
        if (t3[2]*des[3]=t3[3]*des[2])and(t3[3]*des[1]=t3[1]*des[3]) then
          begin
            writeln(i,' ',j,' ',k,' ',sum(t3) div sdes);
            exit;
          end;
      end;
    end;
  end;
  writeln('NONE');
end;
