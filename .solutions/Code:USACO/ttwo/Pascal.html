Program ttwo;
  const
    direction:array[1..4,1..2] of integer=((-1,0),(0,1),(1,0),(0,-1));
  var
    f:array[0..11,0..11] of boolean;
    way:Array[1..10,1..10,1..10,1..10,1..4,1..4] of boolean;
    x1,x2,y1,y2:longint;
    ans:longint;
      procedure init;
        var
          i,j:longint;
          c:string;
        begin
          assign(input,'ttwo.in');
          reset(input);
          assign(output,'ttwo.out');
          rewrite(output);
          fillchar(f,sizeof(f),false);
          fillchar(way,sizeof(way),true);
          ans:=0;
          for i:=1 to 10 do
            begin
              readln(c);
              for j:=1 to 10 do
                begin
                  if c[j]='.' then f[i][j]:=true;
                  if c[j]='F' then
                    begin
                      x1:=i;
                      y1:=j;
                      f[i][j]:=true;
                    end;
                  if c[j]='C' then
                    begin
                      x2:=i;
                      y2:=j;
                      f[i][j]:=true;
                    end;
                end;
            end;
          close(input);
        end;
      procedure work;
        var
          i,j:longint;
        begin
          i:=1;
          j:=1;
          while (x1<>x2)or(y1<>y2) do
            begin
              if way[x1][y1][x2][y2][i][j]=false then
                begin
                  writeln(0);
                  close(output);
                  halt;
                end;
              way[x1][y1][x2][y2][i][j]:=false;
              if (x1=x2)and(y1=y2) then
                begin
                  writeln(ans);
                  close(output);
                  halt;
                end;
              if f[x1+direction[i][1]] [y1+direction[i][2]] then
                begin
                  x1:=x1+direction[i][1];
                  y1:=y1+direction[i][2];
                end else
                  begin
                    i:=i+1;
                    i:=i mod 4;
                    if i=0 then i:=4;
                  end;
              if f[x2+direction[j][1]] [y2+direction[j][2]] then
                begin
                  x2:=x2+direction[j][1];
                  y2:=y2+direction[j][2];
                end else
                  begin
                    j:=j+1;
                    j:=j mod 4;
                    if j=0 then j:=4;
                  end;
              inc(ans);
            end;
        end;
  begin
    init;
    work;
    writeln(ans);
    close(output);
  end.
