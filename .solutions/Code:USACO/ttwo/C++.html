#include <cstdio>
#include <cstdlib>
#include <cstring>
#include <memory.h>
#include <iostream>
#include <algorithm>
using namespace std;
 
const int maxn = 12;
 
char graph[maxn][maxn];
int  fx, fy, cx, cy;
int  dx[] = {-1, 0, 1, 0};
int  dy[] = {0, 1, 0, -1};
 
void ReadData()
{// read the forests
    for ( int i=0; i<10; ++i )
    {
        for ( int j=0; j<10; ++j )
        {
            scanf("%c", &graph[i][j]);
            if ( graph[i][j]=='C' )
                cx=i, cy=j, graph[i][j]='.';
            if ( graph[i][j]=='F' )
                fx=i, fy=j, graph[i][j]='.';
        }
 
		getchar();
    }
}// ReadData
 
bool IsOut( int x, int y )
{
    return ( x<0 || y<0 || x>=10 || y>=10 );
}// IsOut
 
void Simulation()
{
    int  k, fnx, fny, cnx, cny, ti=0;
    int  fstep=0, cstep=0, fdir=0, cdir=0;
 
    while ( fstep<=400 && cstep<=400 )   // 无论对于哪一方来说，地图上最大步数是10*10*4
    {
        if ( fx==cx && fy==cy )
            break;
 
        // Farmer John
        fnx = fx+dx[ fdir ], fny = fy+dy[ fdir ];
        if ( !IsOut( fnx, fny ) && graph[fnx][fny]!='*' )
            fx=fnx, fy=fny, ++fstep;
        else
            fdir = ( fdir+1 )%4;
 
        // Cow
        cnx = cx+dx[ cdir ], cny = cy+dy[ cdir ];
        if ( !IsOut( cnx, cny ) && graph[cnx][cny]!='*' )
            cx=cnx, cy=cny, ++cstep;
        else
            cdir = ( cdir+1 )%4;
 
        ++ti;        // 注意，改变方向需要 1 Unit time，ti 记录最终相遇时间（如果相遇）
    }// End of while
    if ( fstep<=400 && cstep<=400 )
        printf("%d\n", ti);
    else
        printf("0\n");
}// simulation
 
int main()
{
    freopen("ttwo.in", "r", stdin);
    freopen("ttwo.out", "w", stdout);
 
    ReadData();
    Simulation();
 
//    system("pause");
    return 0;
}

#include <fstream>
#include <iostream>
#include <string>
#include <cstring>
using namespace std;
 
const short direction[4][2] = {{-1,0},{0,1},{1,0},{0,-1}};
char map[12][12];
bool used[11][11][4][11][11][4] = {false};
int count;
class T
{
public:
	int x,
		y,
		dire;//up:0 right:1 down:2 left:3
	T() 
	{
		dire = 0;
	}
	void go()
	{
		if(map[direction[dire][0] + x][direction[dire][1] + y] == '*')
			dire = (dire+1)%4;
		else x += direction[dire][0],
			 y += direction[dire][1];
	}
 
};
 
ifstream fin("ttwo.in");
ofstream fout("ttwo.out");
T F,C;
 
void input()
{
	string in;
	for(int i=0; i<=11; i++)
		map[0][i] = '*',
		map[11][i] = '*';
	for(int i=1; i<=10; i++)
	{
		fin >> in;
		map[i][0] = '*';
		map[i][11] = '*';
		for(int j=1; j<=10; j++)
		{
			if(in[j-1] != 'C' && in[j-1] != 'F')
				map[i][j] = in[j-1];
			else 
			{
				map[i][j] = '.';
				if(in[j-1] == 'F')
					F.x = i,
					F.y = j;
				else C.x = i,
					 C.y = j;
			}
 
		}
	}
}
 
int main()
{
	input();
	count = 0;
	while(true)
	{
		if(used[F.x][F.y][F.dire][C.x][C.y][C.dire])
		{
			fout << 0 << endl;
			return 0;
		}
		used[F.x][F.y][F.dire][C.x][C.y][C.dire] = true;
		if(F.x == C.x && F.y == C.y)
		{
			fout << count << endl;
			return 0;
		}
		F.go();
		C.go();
		count++;
	}
    return 0;
}

#include <stdio.h>
#include <stdlib.h>
#include <string.h>
#include <math.h>
#include <assert.h>
#define _DEBUG 0
#define GRID_LENGTH 10
enum Derection{//方向
	NORTH,
	EAST,
	SOUTH,
	WEST
};
int grid[GRID_LENGTH][GRID_LENGTH];//grid,如果为0则为空，1则存在障碍物
bool state[GRID_LENGTH][GRID_LENGTH][4][GRID_LENGTH][GRID_LENGTH][4];//分别表示经过的状态，cow、farmer
typedef struct _Pos{
	int row;//行号
	int column;//列号
	int derect;//方向
}Pos;
inline void setPos(Pos &aPos,int row,int column,int d){//设置位置
	aPos.row=row;
	aPos.column=column;
	aPos.derect=d;
}
bool refresh_stat(Pos cur_cow,Pos cur_farm){//刷新状态
	if(state[cur_cow.row][cur_cow.column][cur_cow.derect][cur_farm.row][cur_farm.column][cur_farm.derect])
		return false;//说明状态已有过，则farmer不能追上cower
	state[cur_cow.row][cur_cow.column][cur_cow.derect][cur_farm.row][cur_farm.column][cur_farm.derect]=true;
	return true;
}
bool check(Pos cur_cow,Pos cur_farm){
	if(cur_cow.row==cur_farm.row && 
		cur_cow.column==cur_farm.column)
		//&&((cur_cow.derect-cur_farm.derect+4)&1)==0)//方向向对		
		return true;
	else
		return false;
}
void move(Pos &obj){
	int i=obj.row;
	int j=obj.column;
	int d=obj.derect;
	bool needRotate=false;
	switch(d){
	case NORTH:
		i=i-1;
		break;
	case EAST:
		j=j+1;
		break;
	case SOUTH:
		i=i+1;
		break;
	case WEST:
		j=j-1;
		break;
	}	
	if(i<GRID_LENGTH && i>=0 &&
		j<GRID_LENGTH && j>=0 &&
		(!grid[i][j])){//如果下一个位置满足条件
		setPos(obj,i,j,d);
	}else//顺时针90度
		obj.derect=(d+1)%4;
}
 
int main(){
	freopen("ttwo.in","r",stdin);
#if _DEBUG==0
	freopen("ttwo.out","w",stdout);
#endif
	int i,j;
	char c;
	Pos cur_cow;//当前奶牛位置
	Pos cur_farm;//当前farmer位置
	cur_cow.derect=cur_farm.derect=NORTH; //都为north方向
	char str[GRID_LENGTH+1];
	for(i=0;i<GRID_LENGTH;++i){
		gets(str);
		for(j=0;j<GRID_LENGTH;++j){
			//scanf("%c",&c);
			c=str[j];
			if(c=='*'){
				grid[i][j]=1;
				continue;
			}
			if(c=='F'){
				setPos(cur_farm,i,j,NORTH);
				continue;
			}
			if(c=='C'){
				setPos(cur_cow,i,j,NORTH);
				continue;
			}
		}
	}
#if _DEBUG==1
	/*

	*/
#endif
	int t=0;//分钟数

#include <iostream>
#include <fstream>
#include <string>
 
using namespace std;
 
struct Grid{
	bool go;	//0: obstacle, 1: F, 2: C
}g[12][12];
 
struct man{
	int x, y;
	int direction;		//0=N, 1=E, 2=S, W=3
}farmer, cow;
 
void move(man &m){
	switch(m.direction){
		case 0:	(g[m.x-1][m.y].go)? m.x-- : m.direction = 1; break;
		case 1:	(g[m.x][m.y+1].go)? m.y++ : m.direction = 2; break;
		case 2:	(g[m.x+1][m.y].go)? m.x++ : m.direction = 3; break;
		case 3:	(g[m.x][m.y-1].go)? m.y-- : m.direction = 0; break;
	}
}
 
 
int main() {
    ofstream fout ("ttwo.out");
    ifstream fin ("ttwo.in");
    char c;
    int minute = 0;
    for (int i = 1 ; i <= 10 ; i++){
    	for (int j = 1 ; j <= 10 ; j++){
    		fin >> c;
    		if (c != '*')	g[i][j].go = true;
    		if (c == 'F')	farmer.x = i, farmer.y = j;
    		if (c == 'C')	cow.x = i, cow.y = j;
    	}
    }
 
    while (!(farmer.x == cow.x && farmer.y == cow.y)){
    	move(farmer);
    	move(cow);
    	minute++;
    	if (minute > 1000000){
    		minute = 0;
    		break;
    	}
    }
 
    fout << minute << endl;
    return 0;
}
