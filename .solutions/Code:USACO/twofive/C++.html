int dp[6][6][6][6][6];
int map[6][6],xx[28],ii[6],used[30];
int dfs(int si,int sj,int res)
{
	if(res>xx[0])
		return dp[ii[1]][ii[2]][ii[3]][ii[4]][ii[5]];
	int now=0;
	if(si+1<=5&&xx[res]>map[si][1]){
		map[si+1][1]=xx[res];
		ii[si+1]++;
		now+=dfs(si+1,1,res+1);
		ii[si+1]--;
		map[si+1][1]=0;
	}
	for(int k=1;k<=si;k++){
		if(k!=1&&ii[k-1]<=ii[k])
			continue;
	if(ii[k]+1<=5&&xx[res]>map[k][ii[k]]&&xx[res]>map[k-1][ii[k]+1]){
		map[k][ii[k]+1]=xx[res];
		ii[k]++;
		now+=dfs(si,sj+1,res+1);
		map[k][ii[k]]=0;
		ii[k]--;
	}
	}
	return now;
}
int Wsum(char* str)
{
	int sum=0,i,j,k;
	memset(used,0,sizeof(used));
	memset(ii,0,sizeof(ii));
	memset(map,0,sizeof(map));
	memset(xx,0,sizeof(xx));
	for(i=0;str[i];i++){
		    for(j=1;j<=str[i]-'A';j++){
				if(used[j])
					continue;
				if(j<map[i/5][i%5+1]||j<map[i/5+1][i%5])
					continue;
				map[i/5+1][i%5+1]=j;
				used[j]=1;
				ii[i/5+1]++;
				int Max=j;
				for(k=0;k<i;k++)
					if(Max<str[k]-'A'+1)
						Max=str[k]-'A'+1;
				xx[0]=0;
				for(k=1;k<=Max;k++)
					if(!used[k])
					    xx[++xx[0]]=k;
				sum+=dfs(i/5+1,i%5+1,1);
				used[j]=0;
				ii[i/5+1]--;
			}
		map[i/5+1][i%5+1]=str[i]-'A'+1;
		ii[i/5+1]++;
		used[str[i]-'A'+1]=1;
	}
	return sum;
}

int calc(int a, int b, int c, int d, int e, int chr){
	int &inwork = dp[a][b][c][d][e];
	if(inwork) return inwork;
	if(used[chr]) return inwork = calc(a, b, c, d, e, chr + 1);
	if(a < 5 && chr > maxr[0] && chr > maxc[a]) inwork += calc(a + 1, b, c, d, e, chr + 1);
	if(b < a && chr > maxr[1] && chr > maxc[b]) inwork += calc(a, b + 1, c, d, e, chr + 1);
