Program prefix(input,output);
  var
    se:array[1..200] of string;
    s:ansistring;
    m:longint;
    f:array[0..200000] of boolean;
      procedure init;
        var
          c:string;
        begin
          assign(input,'prefix.in');
          assign(output,'prefix.out');
          reset(input);
          rewrite(output);
          m:=0;
          while true do
            begin
              readln(c);
              if c='.' then break;
              c:=c+' ';
              while c<>'' do
                begin
                  inc(m);
                  se[m]:=copy(c,1,pos(' ',c)-1);
                  delete(c,1,pos(' ',c));
                end;
            end;
          s:='';
          while not eof do
            begin
              readln(c);
              s:=s+c;
            end;
          close(input);
          fillchar(f,sizeof(f),false);
          f[0]:=true;
        end;
          function check(x:longint;y:string):boolean;
            var
              i:longint;
            begin
              for i:=1 to length(y) do
                if s[i+x]<>y[i] then exit(false);
              exit(true);
            end;
      procedure Dp;
        var
          i,j:longint;
        begin
          for j:=0 to length(s) do
            for i:=1 to m do if length(s)-length(se[i])>=0 then
              if f[j] then
                if check(j,se[i]) then f[j+length(se[i])]:=true;
        end;
      Procedure Out_put;
        var
          j:longint;
        begin
          for j:=length(s) downto 0 do
            if f[j] then
              begin
                writeln(j);
                close(output);
                halt;
              end;
        end;
  Begin
    init;
    Dp;
    Out_put;
  End.
