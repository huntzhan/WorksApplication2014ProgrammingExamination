   J,K	 : Double;
   I,L,G : Longint;
Begin
   G:=0;
   For I:=1 To F Do
      For L:=1 To R Do
      Begin
	 G:=G+1;
	 Grat[G]:=Fs[I]/Rs[L];
      End;
   Isort(G);
   For I:=1 To G-1 Do Cdl[I]:=Grat[I+1]-Grat[I];
   J:=0;
   For I:=1 To G-1 Do J:=J+Cdl[I];
   J:=J/(G-1);
   K:=0;
   For I:=1 To G-1 Do K:=K+(J-Cdl[I])*(J-Cdl[I]);
   If K<H Then
   Begin
      H:=K;
      Move(Fs,Bfs,4*F);
      Move(Rs,Brs,4*R);
   End;
End;
Procedure Fidr(Deep: Longint);
Var
   I : Longint;
Begin
   If Deep>=R Then Cqlt
   Else
      For I:=Rs[Deep-1]+1 To Rs[R]-R+Deep Do
      Begin
	 Rs[Deep]:=I;
	 Fidr(Deep+1);
      End;
End;
Procedure Fidf(Deep: Longint);
Var
   I : Longint;
Begin
   If Deep>=F Then Fidr(2)
   Else
      For I:=Fs[Deep-1]+1 To Fs[F]-F+Deep Do
      Begin
	 Fs[Deep]:=I;
	 Fidf(Deep+1);
      End;
End;
Begin
   Assign(Input,'cowcycle.in');
   Reset(Input);
   Readln(F,R);
   Readln(F1,F2,R1,R2);
   Close(Input);
   H:=Maxlongint;
   For Minf:=F1 To F2 Do
      For Maxf:=Minf+F-1 To F2 Do
	 For Minr:=R1 To R2 Do
	    For Maxr:=Minr+R-1 To R2 Do
	       If (Maxf/Minr>=3*(Minf/Maxr)) Then
	       Begin
		  Fs[1]:=Minf;
		  Fs[F]:=Maxf;
		  Rs[1]:=Minr;
		  Rs[R]:=Maxr;
		  Fidf(2);
	       End;
   Assign(Output,'cowcycle.out');
   Rewrite(Output);
   Write(Bfs[1]);
   For I:=2 To F Do Write(' ',Bfs[I]);
   Writeln;
   Write(Brs[1]);
   For I:=2 To R Do Write(' ',Brs[I]);
   Writeln;
   Close(Output);
End.

procedure DFS(x,y:integer);

            begin
              p[k+1]:=p[k];
              dec(k);
            end;

var
 n,m:longint;
 tot:longint;

 q1,q2,h1,h2:longint;

procedure pai(l,r:longint);
var
 i,j:longint;
 mid,t:extended;
begin
 i:=l;j:=r;
 mid:=f[(l+r)shr 1];
 repeat
  while f[i]<mid do inc(i);
  while f[j]>mid do dec(j);
  if i<=j then
   begin
    t:=f[i];
    f[i]:=f[j];
    f[j]:=t;
    inc(i);
    dec(j);
   end;
 until i>j;
 if i<r then pai(i,r);
 if l<j then pai(l,j);
end;
 
procedure check;
var
 fang:extended;
 i,j:longint;
begin
 sum:=0;fang:=0;tot:=0;
 x:=q[1]/h[m];
 if d/x<3 then exit;
 for i:=1 to n do
  for j:=1 to m do
   begin
    inc(tot);
    f[tot]:=q[i]/h[j];
   end;
 pai(1,tot);
 for i:=2 to tot do begin tt[i-1]:=abs(f[i]-f[i-1]); sum:=sum+tt[i-1]; end;
 sum:=sum/(tot-1);
 for i:=1 to tot-1 do fang:=fang+sqr(tt[i]-sum);
 if fang<min then
  begin
   min:=fang;
   a:=q;b:=h;
  end;
end;
