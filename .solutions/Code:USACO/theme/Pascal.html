    n,i,j,x,y,num,max:integer;

    begin
      read(y);
      a[i]:=y-x;
      x:=y;
    end;
  max:=0;
  for i:=5 to n do
    begin
      if num>max then max:=num;
      num:=0;
      for j:=2 to n-i do
        if (a[j]=a[i+j]) and (num+1<i) then
          inc(num)
        else
          begin
            if num>max then max:=num;
            num:=0;
          end;
    end;
  if max>=4 then writeln(max+1) else writeln(0);
  close(input);
  close(output);
end.

//看看这里没有hash写法的，那么写了一个双hash判重算法，复杂度为O(n*lg(n))，数据弱，用不上。。

uses
  math;
 
const
  p1: int64 = 999997;
  p2: int64 = 999999997;
  base =  7997;
 
var
  n: longint;
  a,b: array [0..6000] of longint;
  bin,f: array [0..6000,1..2]  of int64;
  h: array [0..999997] of longint;
 
procedure init;
var
  i,minv: longint;
begin
  readln(n);
  for i:=1 to n do
    read(a[i]);
  minv:=maxlongint;
  for i:=2 to n do
  begin
    b[i-1]:=a[i]-a[i-1];
    minv:=min(minv,b[i-1]);
  end;
  dec(n);
  for i:=1 to n do
    b[i]:=b[i]-minv;
 
  for i:=1 to n do
    f[i,1]:=((f[i-1,1]*base)+b[i]) mod p1;
  for i:=1 to n do
    f[i,2]:=((f[i-1,2]*base)+b[i]) mod p2;
  bin[0,1]:=1;
  bin[0,2]:=1;
  for i:=1 to n do
    bin[i,1]:=(bin[i-1,1]*base) mod p1;
  for i:=1 to n do
    bin[i,2]:=(bin[i-1,2]*base) mod p2;
end;
 
function hash(st,ed: longint;si: longint): longint;
begin
  if si=1 then
    hash:=(p1*p1+f[ed,1]-f[st-1,1]*bin[ed-st+1,1]) mod p1
  else
    hash:=(p2*p2+f[ed,2]-f[st-1,2]*bin[ed-st+1,2]) mod p2;
end;
 
function judge(len: longint): boolean;
var
  st1,st2: longint;
begin
  fillchar(h,sizeof(h),0);
  h[hash(1,len,1)]:=hash(1,len,2);
  st1:=1;st2:=len+1;
  while (st2+len)<=n do
  begin
    inc(st2);
    if h[hash(st2,st2+len-1,1)]=hash(st2,st2+len-1,2) then
      exit(true);
    inc(st1);
    h[hash(st1,st1+len-1,1)]:=hash(st1,st1+len-1,2);
  end;
  exit(false);
end;
 
procedure work;
var
  l,r,mid: longint;
begin
  l:=0;r:=(n+1) div 2;
  repeat
    mid:=(l+r) div 2;
    if judge(mid) then
      l:=mid
    else
      r:=mid;
  until r-l<=1;
  if r<5 then
    writeln(0)
  else
    writeln(r);
end;
 
begin
  assign(input,'theme.in');reset(input);
  assign(output,'theme.out');rewrite(output);
  init;
  work;
  close(input);close(output);
end.

program ex513;
 const
   maxn = 5000;
   Inf = 100000000;
 var
     n,p,j:         longint;
     len,m:         longint;
     rank,trank,SA,tSA,ws:     array[1..maxn] of longint;
     a,d:         array[1..maxn] of longint;
     Height:      array[1..maxn] of longint;
     Ans:         longint;
 
 function max(a,b:longint):longint;
 
   begin
     if a>b then exit(a) else exit(b);
   end;
 
 function min(a,b:longint):longint;
 
   begin
     if a<b then exit(a) else exit(b);
   end;
 
 
 procedure Init;
   var
       i,min:       longint;
 
   begin
     readln(n);
     for i :=1 to n do read(a[i]);
     for i :=1 to n-1 do d[i] :=a[i+1]-a[i];
     min :=Inf;
     for i :=1 to n-1 do if d[i] < min then min :=d[i];
     for i :=1 to n-1 do d[i] :=d[i] - min + 1;
     len :=n-1;   m :=300;
     for i :=1 to len do  rank[i] :=d[i];
     for i :=1 to len do inc(ws[rank[i]]);
     for i :=2 to m do  inc(ws[i],ws[i-1]);
     for i :=len downto 1 do begin
         SA[ws[rank[i]]] :=i;
         dec(ws[rank[i]]);
         end;
     p :=1;  trank[SA[1]] :=1;
     for i :=2 to len do begin
         if  rank[SA[i-1]] <> rank[SA[i]] then inc(p);
         trank[SA[i]] :=p;
         end;
     m :=p;  rank :=trank;
   end;
 
 
 
 procedure Main;
   var
        i:      longint;
 
   begin
     j :=1;
 
     while m < len do begin
       fillchar(ws,sizeof(ws),0);
       p :=0;
       for i :=len-j+1 to len do begin inc(p); tsa[p] :=i; end;
       for i :=1 to len do if sa[i] > j then begin inc(p); tsa[p] :=sa[i]-j end;
       for i :=1 to len do trank[i] :=rank[tsa[i]];
       for i :=1 to len do inc(ws[trank[i]]);
       for i :=2 to m do inc(ws[i],ws[i-1]);
       for i :=len downto 1 do begin
           sa[ws[trank[i]]] :=tsa[i];
           dec(ws[trank[i]]);
           end;
       p :=1;  trank[sa[1]] :=1;
       for i :=2 to len do begin
           if (rank[sa[i]] <> rank[sa[i-1]]) or
              (rank[sa[i]+j] <> rank[sa[i-1]+j]) then inc(p);
           trank[sa[i]] :=p;
           end;
       m :=p;  j :=j shl 1;  rank :=trank;
       end;   //while
 
    p :=0;
    for i :=1 to len do begin
        if rank[i] = 1 then begin Height[1] :=0; continue end;
        j :=sa[rank[i]-1];
        while d[i+p] = d[j+p] do inc(p);
        Height[rank[i]] :=p;
        if p > 0 then dec(p);
        end;
  end;
 
 
 function can(x:longint):boolean;
   var
       maxsa,minsa,i:      longint;
 
   begin
     maxsa :=-Inf;   minsa :=Inf;
     for i :=1 to len-1 do  begin
         if Height[i+1] >= x then begin
            maxsa :=max(maxsa , sa[i]);
            minsa :=min(minsa , sa[i]);
            maxsa :=max(maxsa , sa[i+1]);
            minsa :=min(minsa , sa[i+1]);
            if maxsa - minsa >x then exit(true);
            end
         else begin
              maxsa :=-Inf;  minsa :=Inf;
              end;
         end;
     exit(false);
   end;
 
 
 
 procedure solve;
   var
       l,r,mid:         longint;
 
   begin
     l :=0;   r :=len;
     while r - l >1 do begin
       mid :=(l+r) shr 1;
       if can(mid) then l :=mid else r :=mid;
       end;
     Ans :=l+1;
     if Ans >=5 then writeln(Ans) else writeln(0);
   end;
 
 
 
 begin
   assign(input,'theme.in');  reset(input);
   assign(output,'theme.out'); rewrite(output);
   Init;
   Main;
   solve;
   close(input);   close(output);
 end.

program theme;
const pr=1335975791;
      maxn=5000;
var a,h:array[1..maxn] of longint;
    s,cj:array[1..maxn] of int64;
    n,i,j,l,r,p,ans:longint;
function work(x:longint):boolean;
var t,i,j,max,min:longint;
procedure qsort(x,y:longint);
var i,j,p,t:longint;
begin
 i:=x;j:=y;p:=s[(i+j) >>1];
 repeat
 while s[i]<p do inc(i);
 while s[j]>p do dec(j);
 if i<=j then
  begin
   t:=s[i];s[i]:=s[j];s[j]:=t;
   t:=h[i];h[i]:=h[j];h[j]:=t;
   inc(i);dec(j);
  end;
 until i>j;
if j>x then qsort(x,j);
if i<y then qsort(i,y);
end;
 
begin
for i:=1 to n-x+1 do
 begin
 s[i]:=1;h[i]:=i;
 t:=i+x-1;
 while t>=i do
  if t-t and (-t)>=i-1 then
  begin
  s[i]:=s[i]*cj[t] mod pr;dec(t,t and -t);
  end
  else begin s[i]:=s[i]*a[t] mod pr;dec(t);end;
 end;
qsort(1,n-x+1);
max:=h[1];min:=h[1];
for i:=1 to n-x do
if (s[i+1]=s[i]) then
 begin
 if h[i+1]>max then max:=h[i+1];
 if h[i+1]<min then min:=h[i+1];
  if (max-min>x) then exit(true);
 end else begin max:=h[i+1];min:=h[i+1];end;
exit(false);
end;
 
begin
assign(input,'theme.in');reset(input);
assign(output,'theme.out');rewrite(output);
readln(n);
for i:=1 to n do read(a[i]);
for i:=1 to n-1 do a[i]:=a[i+1]-a[i]+100;
dec(n);
for i:=1 to n do
 begin
  p:=i and -i;cj[i]:=1;
  for j:=i-p+1 to i do
   cj[i]:=cj[i]*a[j] mod pr;
 end;
l:=1;r:=n;
while l<=r do
 begin
  p:=(l+r)>>1;
  if work(p) then
   begin
    ans:=p;l:=p+1;
   end
  else r:=p-1;
 end;
if ans<4 then writeln(0)
else writeln(ans+1);
close(input);close(output);
end.

  var
    i,j,k,l,m,n:longint;
    a,pre:array[1..5000]of longint;
 
  procedure opf;
    begin
      assign(input,'theme.in');
      reset(input);
      assign(output,'theme.out');
      rewrite(output);
    end;
 
  procedure clf;
    begin
      close(input);
      close(output);
    end;
 
  procedure init;
    begin
      readln(n);
      for i := 1 to n do
        read(a[i]);
      for i := 1 to n-1 do
        a[i]:=a[i]-a[i+1];
    end;

    var
      ans:longint;
    begin
      dec(n);
      ans:=0;
      for i := 1 to n-1 do
        begin
          fillchar(pre,sizeof(pre),0);
          k:=i-1;
          l:=i-1;
          pre[i]:=i-1;
          for j := i+1 to n do
            begin
              while (a[j]<>a[k+1])and(k>i-1) do
                k:=pre[k];
              if a[j]=a[k+1] then inc(k);
              pre[j]:=k;

var a:array[0..10000] of integer;

function find(x:integer):integer;
var be,len:integer;
begin
	find:=0;
    for be:=x+2 to n do
	if a[be]=a[x] then
	begin
		len:=1;
		while (be+len<=n) and (x+len+1<be) and (a[be+len]=a[x+len]) do inc(len);
		if len>find then find:=len;
	end;
end;
begin
assign(input,'theme.in'); assign(output,'theme.out');
reset(input); rewrite(output);

dec(n);

	if i>n-max*2 then break;
