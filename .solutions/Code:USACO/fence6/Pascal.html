program Fence6;
var
  a,f:array [1..100,1..100] of integer; //邻接矩阵
  b:array [1..100,1..2,0..8] of byte; //左右篱笆
  c:array [1..100] of byte; //篱笆长度
  n,s,i,j,ii,jj,k,ans:integer;
begin
  assign(input,'fence6.in');reset(input);
  assign(output,'fence6.out');rewrite(output);
  readln(n);
  for i:=1 to n do begin
    read(s);
    readln(c[s],b[s,1,0],b[s,2,0]);
    for j:=1 to 2 do begin
      for k:=1 to b[s,j,0] do
        read(b[s,j,k]);
      readln;
    end;
  end;
  for i:=1 to n do
    for j:=i to n do begin
      a[i,j]:=10000;
      a[j,i]:=10000;
    end;
  for i:=1 to n do
    for j:=1 to 2 do
      for k:=1 to b[i,j,0] do begin
        a[i,b[i,j,k]]:=c[i]+c[b[i,j,k]];
        a[b[i,j,k],i]:=a[i,b[i,j,k]];
      end;
  f:=a;
  ans:=maxint;
  for k:=1 to n do begin
    for i:=1 to b[k,1,0] do begin
      ii:=b[k,1,i];
      if ii<k then
        for j:=1 to b[k,2,0] do begin
          jj:=b[k,2,j];
          if jj<k then begin
            if f[ii,jj]+a[ii,k]+a[k,jj]<ans then
              ans:=f[ii,jj]+a[ii,k]+a[k,jj];
          end;
        end;
    end;
    for i:=1 to n do
      for j:=1 to n do
        if f[i,k]+f[k,j]<f[i,j] then
          f[i,j]:=f[i,k]+f[k,j];
  end;
  writeln(ans div 2);
  close(input);
  close(output);
end.

			q[t]:=p^.x;
		    end;
		end;
		p:=p^.next;
	    end;
	end;
        ans:=min(ans,len[i]+dis[f[i+n]]);
    end;
    writeln(ans);
    close(input); close(output);
End.
