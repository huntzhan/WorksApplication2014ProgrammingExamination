	if( fp == 0 && dis(a->x,a->y,ST.x,ST.y) < dis(b->x,b->y,ST.x,ST.y)) // 如果在一条直线上，则把远的放在前面

    // calc len
    double len = 0.0;
    for (int i=0; i<top; i++)
        len += dis(r[i], r[(i+1)%top]);
 
    fprintf(fo, "%.2f\n", len);
    fclose(fo);
 
    return 0;
}

void qsort(int l,int r){
    int i=l,j=r;
    double k,x=a[(l+r)/2][0];
    while(i<=j){
        while(a[i][0]<x) i++;
        while(a[j][0]>x) j--;
        if(i<=j) {
          k=a[i][0];a[i][0]=a[j][0];a[j][0]=k;
          k=a[i][1];a[i][1]=a[j][1];a[j][1]=k;
          i++;j--;         
        }            
    }     
    if(i<r) qsort(i,r);
    if(l<j) qsort(l,j);      
}
double xie(int x,int y){
    double p=a[x][0]-a[y][0],q=a[x][1]-a[y][1];
    if(p==0)
       if(q<.0)
          return 10000000;
       else return -10000000;
    else
       q=q/p;
    return q;       
}
double jl(int x,int y){
    double q=a[x][0]-a[y][0],p=a[x][1]-a[y][1];
	q*=q;p*=p;
	q+=p;
	q=sqrt(q);
	return q;     
}
