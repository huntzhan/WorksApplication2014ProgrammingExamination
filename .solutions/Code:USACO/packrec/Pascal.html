  End;

    If a>=b then exit(a);
    exit(b)

            save[i,2]:=temp;
        end;
        sa:=sa div 2;
    end;
end;
 
procedure check(num:integer);
var i,j,maxx,maxy:integer;
begin
    maxx:=0;
    maxy:=0;
    for i:=0 to 15 do
    begin
        turn(i);
        if num=1 then
        begin
            for j:=1 to 4 do
                inc(maxx,save[j,1]);
            for j:=1 to 4 do
                maxy:=max(save[j,2],maxy);
        end
        else if num=2 then
        begin
            maxx:=save[1,1]+save[2,1]+save[3,1];
            if save[4,1]>maxx then maxx:=save[4,1];
            for j:=1 to 3 do
                maxy:=max(save[j,2],maxy);
            inc(maxy,save[4,2]);
        end
        else if num=3 then
        begin
            maxx:=max(save[1,1]+save[2,1],save[3,1])+save[4,1];
            maxy:=max(max(save[1,2],save[2,2])+save[3,2],save[4,2]);
        end
        else if num=4 then
        begin
            maxx:=max(save[1,1],save[2,1])+save[3,1]+save[4,1];
            maxy:=max(save[1,2]+save[2,2],max(save[3,2],save[4,2]));
        end
        else if num=5 then
        {第六种情况，以下约定：1表示左上角，2表示左下角，3表示右上角，4表示右下角,maxx、maxy分别表示外层矩形的横向和纵向长度
         第六种情况最经典问题是重叠。一般情况下一想就会想到矩形横向长度是上下两边最大的，纵向是左右两边最大的，于是一写就成了：
        maxx:=max(save[1,1]+save[3,1],save[2,1]+save[4,1]);
        maxy:=max(save[1,2]+save[2,2],save[3,2]+save[4,2]);
        然后立刻连样例都过不了。
        对样例((1,2),(2,3),(3,4),(4,5))导出计算中数据分析，发现出现问题时4个矩形的摆放依次为((1,2),(3,4),(5,4),(3,2))（自己画图），也就是说，错误的把3号往下重叠了。而此时其外部矩形面积，横向仍旧应为6，纵向应为8，所以此时maxy的表达式应为：
        maxy:=max(save[1,2],save[3,2])+max(save[2,2],save[4,2]);
        显然，后一个表达式和前一个都不通用。那什么时候应该用前一个表达式？观察得到的计算中数据可以得出(save[2,2]>save[4,2])and(save[3,1]>save[4,1])为真时需用前一个。
        但是，等一下，请把这个图形上下翻转一下——1和2互换，3和4互换，然后可以发现，我们还需要补充条件——这样又可以得到(save[1,1]>save[2,1])and(save[4,2]>save[2,2])为真时符合用前一个的条件。
        如何安排这些条件？出现一种就要用后一个式子，所以用or连接，于是得到以下布尔表达式
        (((save[1,1]>save[2,1])and(save[4,2]>save[2,2]))or((save[2,2]>save[4,2])and(save[3,1]>save[4,1])))
        这样，问题就解决了
 
        补充：为什么不考虑横向，而只对加纵向是考虑？横向这样考虑时决定了最大宽度，并没发生重叠，如果有，纵向移动即可。同样，你可以先加纵向，然后横向移动}
        begin
            maxx:=max(save[1,1]+save[3,1],save[2,1]+save[4,1]);
            if (((save[1,1]>save[2,1])and(save[4,2]>save[2,2]))or((save[2,2]>save[4,2])and(save[3,1]>save[4,1]))) then maxy:=max(save[1,2],save[3,2])+max(save[2,2],save[4,2])
            else maxy:=max(save[1,2]+save[2,2],save[3,2]+save[4,2]);
        end;
        if maxx*maxy<=minre then ready(maxx,maxy);
        turn(i);
        maxx:=0;
        maxy:=0;
    end;
end;
