					C->num[j]--;
					if(C->num[j] >= 0) break;
					else C->num[j] += 10;
				}
			}
			C->num[i] -= B.num[i];
        	}
		if(!C->num[LEN]) C->len;
        	return  *C;
    	}

        	len = A.len ;
