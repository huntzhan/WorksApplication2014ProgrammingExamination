procedure link(a1,a2,a3,a4,a5:integer);
var tt:shortint;
begin
    if (map[a1,a2,a3,a4,a5,0]=0) then
        for tt:=1 to n do
            if (a1>=discount[tt,1])and(a2>=discount[tt,2])and(a3>=discount[tt,3])and
            (a4>=discount[tt,4])and(a5>=discount[tt,5])then
            begin
                inc(map[a1,a2,a3,a4,a5,0]);
                map[a1,a2,a3,a4,a5,map[a1,a2,a3,a4,a5,0]]:=tt;
                link(a1-discount[tt,1],a2-discount[tt,2],a3-discount[tt,3],
                a4-discount[tt,4],a5-discount[tt,5]);
            end;
end;
