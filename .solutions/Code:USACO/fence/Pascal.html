program fence;
 
var
 
  ans: array[1..1024] of integer;
  edge: array[1..500, 1..500] of integer; {边}
  t: array[1..500] of integer;  {度}
  max: integer = 0;
  v: integer = 0;
  f, k: integer;
  i, e, s: integer;
 
  function startpoint: integer; {获取起始点}
  var
    i, s, e: integer;
  begin
    startpoint := 1;
    for i := 1 to max do
      if odd(t[i]) then
        exit(i);
  end;
 
  procedure dfs(p: integer);
  var
    i: integer;
  begin
    for i := 1 to max do
      if edge[p, i] > 0 then
      begin
        Dec(edge[p, i]);
        Dec(edge[i, p]);
        dfs(i);
      end;
    v := v + 1;
    ans[v] := p;
  end;
 
begin
  Assign(input, 'fence.in');
  reset(input);
  Assign(output, 'fence.out');
  rewrite(output);
  readln(f);
  for i := 1 to f do
  begin
    readln(s, e);
    Inc(t[s]);
    Inc(t[e]);
    Inc(edge[s, e]);
    Inc(edge[e, s]);
    if s > max then
      max := s;
    if e > max then
      max := e;
  end;
  k := startpoint;
  dfs(k);
  for i := v downto 1 do
    writeln(ans[i]);
  Close(output);
end.
