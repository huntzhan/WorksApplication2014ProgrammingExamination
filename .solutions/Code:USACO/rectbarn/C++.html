FILE *fin, *fout;
const int maxn = 3005, maxp = 30005;
bool able[maxn], visit[maxn];
int dx[maxp], dy[maxp];
int st[maxn], ed[maxn], up[maxn];
int n, m, p, ans;
 
void init()
{
	fin = fopen("rectbarn.in", "r");
	fscanf(fin, "%d %d %d", &n, &m, &p);
	for (int i = 1; i <= p; i++){
		fscanf(fin, "%d %d", &dx[i], &dy[i]);
	}
	fclose(fin);
}
 
void quick_sort(int l, int r)
{
	int i = l, j = r, m = dx[(l + r) >> 1];
	int t;
	while (i <= j){
		while (dx[i] < m) i++;
		while (dx[j] > m) j--;
		if (i <= j){
			t = dx[i]; dx[i] = dx[j]; dx[j] = t;
			t = dy[i]; dy[i] = dy[j]; dy[j] = t;			
			i++; j--;
		}
	}
	if (i < r) quick_sort(i, r);
	if (j > l) quick_sort(l, j);
}
 
void compete()
{
	int i;
	for (i = 1; i <= p; i++)
		if (dx[i] != dx[i-1]){
			st[dx[i]] = i; ed[dx[i-1]] = i-1;
		}
	ed[dx[p]] = p;
}
 
void work()
{
	quick_sort(1, p);
	compete();
	int i, j, k, L, R;
	for (i = 1; i <= n; i++){
		for (j = 1; j <= m; j++) able[j] = 1, visit[j] = 1;
		for (j = st[i]; j <= ed[i]; j++) able[dy[j]] = 0;
		for (j = 1; j <= m; j++)
			if (able[j]) up[j]++; else up[j] = 0;
		for (j = 1; j <= m; j++) if (up[j] <= up[j-1]) visit[j] = 0;
		for (L = 1; L <= m; L++) if (able[L] && visit[L]){
			k = up[L];
			if (k * (m - L + 1) <= ans) continue;
			for (R = L; R <= m; R++){
				if (!able[R] || up[R] <= up[L - 1]) break;
				if (up[R] < k) k = up[R];
				if ((R - L + 1) * k > ans) ans = (R - L + 1) * k;
			}
		}
	}
}

}

int compare(const void* p1,const void* p2)
{
    return ( ((Node*)p1)->x- ((Node*)p2)->x )*Maxl+( ((Node*)p1)->y- ((Node*)p2)->y );
} /* qsort 比较函数 先按横坐标再按纵坐标 */

		map[x][y] = 1;
	}
	for(int i = 1; i <= r; i++){
		p = 0; //统计横向可以扩展的长度
		for(int j = 1; j <= c; j++)
			if(!map[i][j]){
				h[j] = h[j] + 1, ++p;
				lr[j][0] = MIN(lr[j][0], p);
			}else
				h[j] = 0, lr[j][0] = 0, p = 0;
		p = 0;
		for(int j = c; j >= 1; j--)
			if(!map[i][j]){
				++p, lr[j][1] = MIN(lr[j][1], p);
				ans = MAX(ans, h[j] * (lr[j][0] + lr[j][1] - 1));
			}else
				lr[j][1] = 0, p = 0;
	}
	cout << ans << endl;
	return 0;
}
