program rectbarn;
var i,i1,j,ii,jj,n,m,p,a,b,ans,t:longint;
    pd:array[0..3001,0..3001]of boolean;
    left,right,high:array[0..1,0..3001]of integer;
procedure init;
begin
    assign(input,'rectbarn.in');reset(input);
    assign(output,'rectbarn.out');rewrite(output);
    readln(n,m,p); ans:=0;
    fillchar(pd,sizeof(pd),true);
    for i:=0 to n+1 do
    begin
        pd[i,0]:=false;
        pd[i,m+1]:=false;
    end;
    for i:=0 to m+1 do
    begin
        pd[0,i]:=false;
        pd[n+1,i]:=false;
    end;
    for i:=1 to p do
    begin
        readln(a,b);
        pd[a,b]:=false;
    end;
end;
procedure main;
begin
    i:=0;
    for ii:=1 to n do
    begin
      i1:=i; i:=1-i;
      for j:=m+1 downto 1 do
      begin
        //right[i,j]:=0;
        if not pd[ii,j] then right[i,j]:=j
          else right[i,j]:=right[i,j+1];
      end;
      for j:=0 to m do
      begin
        //left[i,j]:=0;
        if not pd[ii,j] then left[i,j]:=j
          else left[i,j]:=left[i,j-1];
      end;
      for j:=1 to m do
        if pd[ii,j] then
        begin
            if pd[ii-1,j] then
            begin
                high[i,j]:=high[i1,j]+1;
                if left[i1,j]>left[i,j] then left[i,j]:=left[i1,j];
                if right[i1,j]<right[i,j] then right[i,j]:=right[i1,j];
            end
            else high[i,j]:=1;
            t:=(right[i,j]-left[i,j]-1)*high[i,j];
            if t>ans then ans:=t;
        end;
    end;
end;
procedure print;
begin
    writeln(ans);
    close(input);close(output);
end;
begin
    init;
    main;
    print;
end.

program dfnsk;
uses math;
const OO=10000000;
var
  broke:array[1..3000,1..3000] of boolean;
  f,l,r,left,right:array[0..3001] of longint;
  i,j,m,n,p,x,y,ans:longint;
begin
  assign(input,'rectbarn.in');reset(input);
  assign(output,'rectbarn.out');rewrite(output);
  readln(n,m,p);
  for i:=1 to p do
  begin
    readln(x,y);
    broke[y,x]:=true;//郁闷...改成broke[x,y]就WA了
  end;
  ans:=1;
  for i:=1 to m do
  begin
    left[0]:=0;right[n+1]:=0;
    for j:=1 to n do
      if broke[i,j] then left[j]:=0 else left[j]:=left[j-1]+1;
    for j:=n downto 1 do
      if broke[i,j] then right[j]:=0 else right[j]:=right[j+1]+1;
    for j:=1 to n do
    begin
      if broke[i,j] then
      begin
        f[j]:=0;
        l[j]:=OO;
        r[j]:=OO;
      end else
      begin
        inc(f[j]);
        l[j]:=min(l[j],left[j]);
        r[j]:=min(r[j],right[j]);
      end;
      ans:=max(ans,f[j]*(l[j]+r[j]-1));
    end;
  end;
  if p=0 then writeln(n*m) else writeln(ans);
  close(input);close(output);
end.

program rectbarn;
  type
    ty=record
      x,y:integer;
     end;
  var
    l,r,h:array[1..3100]of integer;
{分别在i行时存储j列所能达到的左边界，右边界，和上边界h[j]表示到达j时的最大高度，此处的l,r,h会随i的变化而变化}
    aa:array[1..31000]of ty;{存储不能利用的坐标}
    n,m,k:integer;
    max:longint;{最后结果}
  procedure  int;
    var
      i:integer;
    begin
      assign(input,'rectbarn.in');
      reset(input);
      readln(n,m,k);
      for i:=1 to k do
        readln(aa[i].x,aa[i].y);
      close(input);
    end;
  procedure  kuai(s,t:integer;var ii:integer);
    var
      temp:ty;
      i,j:integer;
    begin
      i:=s;j:=t;temp:=aa[s];
      while i<j do
        begin
          while (i<j)and((aa[j].x>temp.x)or((aa
 
[j].x=temp.x)and(aa[j].y>=temp.y))) do
            dec(j);
          aa[i]:=aa[j];
          while (i<j)and((aa[i].x<temp.x)or((aa
 
[i].x=temp.x)and(aa[i].y<=temp.y))) do
            inc(i);
          aa[j]:=aa[i];
        end;
      aa[i]:=temp;
      ii:=i;
    end;
  procedure  pai(s,t:integer);{快排}
    var
      ii:integer;
    begin
      if s<t then  //注意此处不要写成while s<t do 要不然成死循环，我就犯了这样的低级错误
        begin
          kuai(s,t,ii);
          pai(s,ii-1);
          pai(ii+1,t);
        end;
    end;
  procedure  check(s,t:integer);//计算过程，注意一定要体味此过程。得高人指点
    var
      i:integer;
      temp:longint;
    begin
      for i:=s to t do//枚举计算此中包含的矩阵
        begin
          inc(h[i]);
          if h[i]=1 then//如果高度为1，则此矩阵的square为1*(t-s+1)
            begin
              l[i]:=s;
              r[i]:=t;
            end
          else
            begin
              if s>l[i] then l[i]:=s;//检查更新左右边界
              if t<r[i] then r[i]:=t;
            end;
          temp:=h[i]*(r[i]-l[i]+1);
          if temp>max then max:=temp;
        end;
     end;
  procedure  main;//注意体味此过程
    var
      i,j,newl,p:integer;
    begin
      max:=0;p:=1;
      fillchar(h,sizeof(h),0);//高度赋初始值
      {for i:=1 to m do
        begin
          l[i]:=1;
          r[i]:=m;
        end; }
      for i:=1 to n do
        begin
          newl:=1;    //左边界赋初始值
          for j:=1 to m do
            if (i=aa[p].x)and(j=aa[p].y) then
              begin
                check(newl,j-1);
                newl:=j+1;//更新左边界
                h[j]:=0;//此时，j列的高度变为0
                while (aa[p].y=j)and(aa[p].x=i) do
                  inc(p);
              end;
          check(newl,m);//计算i行时，最右边的不能用的点到右边界
        end;
    end;
  procedure  out1;
    begin
      assign(output,'rectbarn.out');
      rewrite(output);
     { for i:=1 to k do//检查快排是否正确（调试信息，请忽略 －－aLeN.PN注）
         write(aa[i].x:4);
       writeln;
       for i:=1 to k do
         write(aa[i].y:4);
      writeln;}
      writeln(max);
      close(output);
    end;
begin
  int;
  pai(1,k);
  main;
  out1;

program djy;
var
  map:array[0..3000,0..3000] of boolean;
  h,l,r,ll,rr,lasth,lastl,lastr,lastll,lastrr:array[0..3000] of integer;
  n,m,p,ans:longint;
procedure init;
var
  i,x,y:longint;
begin
  readln(n,m,p);
  for i:=1 to p do
  begin
    readln(x,y);
    map[x,y]:=true;
  end;
end;
procedure main;
var
  i,j:longint;
begin
  for i:=1 to n do
  begin
    for j:=1 to m do
    if map[i,j]=false then h[j]:=lasth[j]+1
    else h[j]:=0;
    for j:=1 to m do
    if map[i,j]=false then
    begin
      if (j=1) or (map[i,j-1]=true) then l[j]:=1
      else l[j]:=l[j-1]+1;
      ll[j]:=l[j];
      if (i>1) and (map[i-1,j]=false) and (lastll[j]<ll[j]) then ll[j]:=lastll[j];
    end;
    for j:=m downto 1 do
    if map[i,j]=false then
    begin
      if (j=m) or (map[i,j+1]=true) then r[j]:=1
      else r[j]:=r[j+1]+1;
      rr[j]:=r[j];
      if (i>1) and (map[i-1,j]=false) and (lastrr[j]<rr[j]) then rr[j]:=lastrr[j];
    end;
    for j:=1 to m do
    if (map[i,j]=false) and (h[j]*(ll[j]+rr[j]-1)>ans) then ans:=h[j]*(ll[j]+rr[j]-1);
    lasth:=h;
    lastl:=l;
    lastr:=r;
    lastll:=ll;
    lastrr:=rr;
  end;
end;
procedure print;
begin
  writeln(ans);
end;
begin
  assign(input,'rectbarn.in');
  reset(input);
  assign(output,'rectbarn.out');
  rewrite(output);
  init;
  main;
  print;
  close(input);
  close(output);
end.
