const
	MAXN = 100;
 
type
	arrn = array[1..MAXN] of longint;
 
var
	M,i,N,x,y,p,node,j : longint;
	hash : array[1..MAXN] of boolean;
	dist : arrn;
	map : array[1..MAXN] of arrn;

	assign(output,'concom.out'); rewrite(output);

		readln(x,y,p); map[x,y] := p;
		N := max(N,x); N := max(N,y);
	end;
