#include<iostream>
using namespace std;
int d[55][55];
int main()
{
    int r,c,i,j,k,sum,max=0,mark=0;
    int a[8];
    cin>>r>>c;
    memset(d,-1,sizeof(d));
    for(i=1;i<=r;i++)
    for(j=1;j<=c;j++)
    scanf("%d",&d[i][j]);
    for(i=1;i<=r+1;i++)
    for(j=1;j<=c+1;j++)
    {
        if(d[i][j]==0) continue;
        sum=0;
        a[0]=d[i-1][j-1];a[3]=d[i][j];a[5]=d[i-1][j+1];
        a[1]=d[i-1][j];a[4]=d[i][j+1];a[6]=d[i+1][j];
        a[2]=d[i-1][j+1];a[7]=d[i+1][j+1];
        for(k=0;k<8;k++)
        {
            if(a[k]==-1||a[k]==0) continue;
            if(sum>=2) 
            {
                mark=d[i][j];
                break;
            }
            if(d[i][j]==a[k]) sum++;
        }
        if(mark>=max) max=mark;
    }
    printf("%d\n",max);
    return 0;
}
