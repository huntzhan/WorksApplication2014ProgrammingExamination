{

    for (int i = 0; i < n; ++i) {
	for (int j = 0; j < n; ++j) {
	    filein >> dist[i][j];

	inTree[i] = false;
    }
    filein.close ();

    distToTree[0] = 0;

	int     best = -1;
	for (int j = 0; j < n; ++j) {
	    if (!inTree[j]) {
		if (best == -1 || distToTree[best] > distToTree[j]) {
		    best = j;
		}
	    }
	}
	assert (best != -1);
	assert (!inTree[best]);
	assert (distToTree[best] < BIG);

	cost += distToTree[best];
	distToTree[best] = 0;
	for (int j = 0; j < n; ++j) {
	    if (distToTree[j] > dist[best][j]) {
		distToTree[j] = dist[best][j];
		assert (!inTree[j]);
	    }
	}
    }
    ofstream fileout ("agrinet.out");
