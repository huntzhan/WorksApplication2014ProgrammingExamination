#include <cstdio>
#include <cstring>

const int maxn=51234;

struct jilu
{
	int add,left,right,max,maxl,maxr;
}tree[maxn*4];

int c[maxn],n,m,d,ans,k;

int Max(int x,int y,int z)
{
	if ( x>=y && x>=z) return x;
	if ( y>=x && y>=z) return y;
	if ( z>=x && z>=y) return z;
}

void Build(int p,int l,int r)
{
	tree[p].left=l;
	tree[p].right=r;
	tree[p].add=0;
	tree[p].max=tree[p].maxl=tree[p].maxr=r-l+1;
	if (l<r)
	{
		int mid=(l+r) >> 1;
		Build(p << 1,l,mid);
		Build((p << 1)+1,mid+1,r);
	}
	
}

void find(int p,int k)
{
	if (tree[p].add==0)
	{
		tree[p << 1].add=0;
		tree[(p << 1)+1].add=0;
		tree[p].max=tree[p].maxl=tree[p].maxr=tree[p].right-tree[p].left+1;
		tree[p].add=-1;
		tree[p << 1].max=tree[p << 1].maxl=tree[p << 1].maxr=tree[p << 1].right-tree[p << 1].left+1;
		tree[(p << 1)+1].max=tree[(p << 1)+1].maxl=tree[(p << 1)+1].maxr=tree[(p << 1)+1].right-tree[(p << 1)+1].left+1;
	}
	if (tree[p].add==1)
	{
		tree[p << 1].add=1;
		tree[(p << 1)+1].add=1;
		tree[p].max=tree[p].maxl=tree[p].maxr=0;
		tree[p].add=-1;
		return ;
	}
	if (tree[p].max<k) return;
	if (tree[p << 1].max>=k) find(p << 1,k);
	else
	if (tree[p << 1].maxr+tree[(p << 1)+1].maxl>=k)
	{
		ans=tree[p << 1].right-tree[p << 1].maxr+1;
	}
	else find((p << 1)+1,k);
}

void change(int p,int l,int r)
{
	if (tree[p].left!=tree[p].right)
	{
	if (tree[p].add==0)
	{
		tree[p << 1].add=0;
		tree[(p << 1)+1].add=0;
		tree[p].max=tree[p].maxl=tree[p].maxr=tree[p].right-tree[p].left+1;
		tree[p].add=-1;
		tree[p << 1].max=tree[p << 1].maxl=tree[p << 1].maxr=tree[p << 1].right-tree[p << 1].left+1;
		tree[(p << 1)+1].max=tree[(p << 1)+1].maxl=tree[(p << 1)+1].maxr=tree[(p << 1)+1].right-tree[(p << 1)+1].left+1;
	}
	if (tree[p].add==1)
	{
		tree[p << 1].add=1;
		tree[(p << 1)+1].add=1;
		tree[p].max=tree[p].maxl=tree[p].maxr=0;
		tree[p].add=-1;
		tree[p << 1].max=tree[p << 1].maxl=tree[p << 1].maxr=0;
		tree[(p << 1)+1].max=tree[(p << 1)+1].maxl=tree[(p << 1)+1].maxr=0;
	}
	}
	if (tree[p].left==l && tree[p].right==r)
	{
		if (k!=0)
		{
			tree[p].max=tree[p].maxl=tree[p].maxr=0;
			tree[p].add=1;
		}
		else
		{
			tree[p].max=tree[p].maxl=tree[p].maxr=tree[p].right-tree[p].left+1;
			tree[p].add=0;
		}
		return;
	}
	else
	{
		int mid=(tree[p].left+tree[p].right) >>1;
		if (r<=mid) 
		{
			change(p << 1,l,r);
			if (tree[p << 1].add!=tree[p].add)
			tree[p].add=-1;
		}
		else 
		if ( l>mid ) 
		{
			change((p << 1)+1,l,r);
			if (tree[p].add!=tree[(p << 1)+1].add)
			tree[p].add=-1;
		}
		else
		{
			change(p << 1,l,mid);
			change((p << 1)+1,mid+1,r);
			if (tree[p << 1].add!=tree[(p << 1)+1].add)
			tree[p].add=-1;
			else
			tree[p].add=tree[p << 1].add;
		}
			tree[p].maxl=tree[p << 1].maxl;
			if (tree[p].maxl==tree[p << 1].right-tree[p << 1].left+1)
			tree[p].maxl+=tree[(p << 1)+1].maxl;
			tree[p].maxr=tree[(p << 1)+1].maxr;
			if (tree[p].maxr==tree[(p << 1)+1].right-tree[(p << 1)+1].left+1)
			tree[p].maxr+=tree[p << 1].maxr;
			tree[p].max=Max(tree[p << 1].max,tree[(p << 1)+1].max,tree[p << 1].maxr+tree[(p << 1)+1].maxl);
	}
}
